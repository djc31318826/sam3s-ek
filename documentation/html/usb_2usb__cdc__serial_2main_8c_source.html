00032  *
00033  * \section Purpose
00034  *
00035  * The USB CDC Serial Project will help you to get familiar with the
00036  * USB Device Port(UDP) and USART interface on SAM microcontrollers. Also
00037  * it can help you to be familiar with the USB Framework that is used for
00038  * rapid development of USB-compliant class drivers such as USB Communication
00039  * Device class (CDC).
00040  *
00041  * You can find following information depends on your needs:
00042  * - Sample usage of USB CDC driver and USART driver.
00043  * - USB CDC driver development based on the USB Framework.
00044  * - USB enumerate sequence, the standard and class-specific descriptors and
00045  *   requests handling.
00046  * - The initialize sequence and usage of UDP interface.
00047  * - The initialize sequence and usage of USART interface with PDC.
00048  *
00049  * \section See
00050  * - usart: USART interface driver
00051  * - tc: TIMER/COUNTER interface driver
00052  * - usb: USB Framework, USB CDC driver and UDP interface driver
00053  *    - \ref usbd_framework
00054  *       - \ref usbd_api
00055  *    - \ref usbd_cdc
00056  *       - \ref usbd_cdc_serial_drv
00057  *       - \ref usbd_cdc_host_drv
00058  *
00059  * \section Requirements
00060  *
00061  * This package can be used with SAM3S evaluation kits.
00062  *
00063  * \section Description
00064  *
00065  * When an EK running this program connected to a host (PC for example), with
00066  * USB cable, the EK appears as a Seriao COM port for the host, after driver
00067  * installation with the offered 6119.inf. Then the host can send or receive
00068  * data through the port with host software. The data stream from the host is
00069  * then sent to the EK, and forward to USART port of SAM chips. The USART
00070  * port of the EK is monitored by the timer and the incoming data will be sent
00071  * to the host.
00072  *
00073  * \section Usage
00074  *
00075  * -# Build the program and download it inside the evaluation board. Please
00076  *    refer to the
00077  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6421.pdf">
00078  *    SAM-BA User Guide</a>, the
00079  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00080  *    GNU-Based Software Development</a> application note or to the
00081  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00082  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00083  * -# On the computer, open and configure a terminal application
00084  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00085  *   - 115200 bauds
00086  *   - 8 bits of data
00087  *   - No parity
00088  *   - 1 stop bit
00089  *   - No flow control
00090  * -# Start the application.
00091  * -# In the terminal window, the following text should appear:
00092  *     \code
00093  *     -- USB Device CDC Serial Project xxx --
00094  *     -- xxxxxx-xx
00095  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00096  *     \endcode
00097  * -# When connecting USB cable to windows, the LED blinks, and the host
00098  *    reports a new USB %device attachment (if it's the first time you connect
00099  *    an %audio speaker demo board to your host). You can use the inf file
00100  *    libraries\\usb\\device\\cdc-serial\\drv\\6119.inf to install the serial
00101  *    port. Then new "SAM3S USB to Serial Converter (COMx)" appears in the
00102  *    hardware %device list.
00103  * -# You can run hyperterminal to send data to the port. And it can be seen

00313 /**
00314  * Handles interrupts coming from Timer #0.
00315  */
00316 void TC0_IrqHandler(void)
00317 {
00318     Tc *pTc0 = TC0;
00319     uint8_t size;
00320     volatile uint32_t status = pTc0->TC_CHANNEL[0].TC_SR;
00321 
00322     if ((status & TC_SR_CPCS) != 0) {
00323 
00324         tcTick ++;
00325 
00326         if (isCdcSerialON) {
00327 
00328             /* Flush PDC buffer */
00329             size = DATABUFFERSIZE - BASE_USART->US_RCR;
00330             if (size == 0) {
00331 
00332                 pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00333                 return;
00334             }
00335             BASE_USART->US_RCR = 0;
00336 
00337             /* Debug log: US->USB */
00338             _DebugLog(hdrS2U, 3, usartBuffers[usartCurrentBuffer], size);
00339             //_DebugLog("\n\r>", 3, usartBuffers[usartCurrentBuffer], size);
00340 
00341             /* Send current buffer through the USB */
00342             while (CDCDSerialDriver_Write(usartBuffers[usartCurrentBuffer],
00343                                           size, 0, 0) != USBD_STATUS_SUCCESS);
00344 
00345             /* Restart read on buffer */
00346             USART_ReadBuffer(BASE_USART,
00347                              usartBuffers[usartCurrentBuffer],
00348                              DATABUFFERSIZE);
00349             usartCurrentBuffer = 1 - usartCurrentBuffer;
00350         }
00351         pTc0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00352     }
00353 }

00404 /*----------------------------------------------------------------------------
00405  * Handles interrupts coming from USART
00406  *----------------------------------------------------------------------------*/
00407 void USART_Handler(void)
00408 {
00409     uint32_t status;
00410     unsigned short serialState;
00411 
00412     status  = BASE_USART->US_CSR;
00413     status &= BASE_USART->US_IMR;
00414 
00415     /* If USB device is not configured, do nothing */
00416     if (!isCdcSerialON) {
00417 
00418         BASE_USART->US_IDR = 0xFFFFFFFF;
00419         return;
00420     }
00421 
00422     /* Buffer has been read successfully */
00423     if ((status & US_CSR_ENDRX) != 0) {
00424 
00425         /* Disable timer */
00426         TC_Stop(TC0, 0);
00427 
00428         /* Debug Log: US->USB */
00429         _DebugLog(hdrS2U, 3,
00430                  usartBuffers[usartCurrentBuffer], DATABUFFERSIZE);
00431         //_DebugLog("\n\r>", 3,
00432         //         usartBuffers[usartCurrentBuffer], DATABUFFERSIZE);
00433 
00434         /* Send buffer through the USB */
00435         while (CDCDSerialDriver_Write(usartBuffers[usartCurrentBuffer],
00436                                 DATABUFFERSIZE, 0, 0) != USBD_STATUS_SUCCESS);
00437 
00438         /* Restart read on buffer */
00439         USART_ReadBuffer(BASE_USART,
00440                          usartBuffers[usartCurrentBuffer],
00441                          DATABUFFERSIZE);
00442         usartCurrentBuffer = 1 - usartCurrentBuffer;
00443 
00444         /* Restart timer */
00445         TC_Start(TC0, 0);
00446     }
00447 
00448     /* Buffer has been sent */
00449     if ((status & US_IER_TXBUFE) != 0) {
00450 
00451         /* Restart USB read */
00452         CDCDSerialDriver_Read(usbBuffer,
00453                               DATABUFFERSIZE,
00454                               (TransferCallback) _UsbDataReceived,
00455                               0);
00456         BASE_USART->US_IDR = US_IER_TXBUFE;
00457     }
00458 
00459     /* Errors */
00460     serialState = CDCDSerialDriver_GetSerialState();
00461 
00462     /* Overrun */
00463     if ((status & US_CSR_OVRE) != 0) {
00464 
00465         TRACE_WARNING( "USART1_IrqHandler: Overrun\n\r");
00466         serialState |= CDCSerialState_OVERRUN;
00467     }
00468 
00469     /* Framing error */
00470     if ((status & US_CSR_FRAME) != 0) {
00471 
00472         TRACE_WARNING( "USART1_IrqHandler: Framing error\n\r");
00473         serialState |= CDCSerialState_FRAMING;
00474     }
00475 
00476     CDCDSerialDriver_SetSerialState(serialState);
00477 }
