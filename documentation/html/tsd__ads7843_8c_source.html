00101  * \brief Timer handler for touch screen. Increments the timestamp counter.
00102  * Determine the state "Pen Pressed" or "Pen Released". To change state,
00103  * the penIRQ has to keep the same value during DEBOUNCE_TIME.

00122             TSD_GetRawMeasurement(data);
00123             TSDCom_InterpolateMeasurement(data, point);

00133         }

00137     if ( PIO_Get( &pinPenIRQ ) == 0 )
00138     {
00139         /* reinit the last time when release */
