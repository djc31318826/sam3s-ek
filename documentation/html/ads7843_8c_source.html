00082 /**
00083  * \brief Generic function to send a command to the touchscreen controller.
00084  *
00085  * \param bCmd command to send
00086  *
00087  * \return command result
00088  */
00089 static uint32_t SendCommand( uint8_t bCmd )
00090 {
00091     uint32_t  uResult = 0;
00092     uint32_t  uTimeout = 0;
00093 
00094     /* (volatile declaration needed for code optimisation by compiler) */
00095     volatile uint8_t bufferRX[3];
00096     volatile uint8_t bufferTX[3];
00097 
00098     Pdc *pPdc = (Pdc *)SPI;
00099     uint32_t dStatus;
00100 
00101     bufferRX[0] = 0;
00102     bufferRX[1] = 0;
00103     bufferRX[2] = 0;
00104 
00105     bufferTX[0] = bCmd;
00106     bufferTX[1] = 0;
00107     bufferTX[2] = 0;
00108 
00109     /* Send Command and data through the SPI */
00110     pPdc->PERIPH_PTCR = PERIPH_PTCR_RXTDIS;
00111     pPdc->PERIPH_RPR = (uint32_t) bufferRX;
00112     pPdc->PERIPH_RCR = 3;
00113 
00114     pPdc->PERIPH_PTCR = PERIPH_PTCR_TXTDIS;
00115     pPdc->PERIPH_TPR = (uint32_t) bufferTX;
00116     pPdc->PERIPH_TCR = 3;
00117 
00118     pPdc->PERIPH_PTCR = PERIPH_PTCR_RXTEN;
00119     pPdc->PERIPH_PTCR = PERIPH_PTCR_TXTEN;
00120 
00121     do
00122     {
00123         dStatus = REG_SPI_SR;
00124         uTimeout++;
00125     }
00126     while ((( dStatus & SPI_SR_RXBUFF) != SPI_SR_RXBUFF) && (uTimeout < ADS7843_TIMEOUT));
00127 
00128     pPdc->PERIPH_PTCR = PERIPH_PTCR_RXTDIS;
00129     pPdc->PERIPH_PTCR = PERIPH_PTCR_TXTDIS;
00130 
00131     uResult = (uint32_t)bufferRX[1] << 8;
00132     uResult |= (uint32_t)bufferRX[2];
00133     uResult = uResult >> 4;
00134 
00135     return uResult;
