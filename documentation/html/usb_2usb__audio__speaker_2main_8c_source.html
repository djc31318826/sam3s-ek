00248  * Configure the TC0 and DACC for audio output.
00249  * \param sampleRate Audio sample rate.
00250  * \param nbChannels Number of audio channels.
00251  * \param mck        MCK frequence.
00252  */

00254                                uint8_t nbChannels,
00255                                uint32_t mck)
00256 {
00257     uint32_t div = 2;
00258     uint32_t tcclks = TC_CMR_TCCLKS_TIMER_CLOCK1;

00260     double   ra, rc;
00261

00265         uint32_t divs[5] = {2, 8, 32, 128,BOARD_MCK / 32768};
00266         uint8_t i = 0;
00267         divs[4] = mck/32768;
00268         /* Minimize DIV & Maxmize RC for better waveform */
00269         while(freq < ((mck / divs[i]) / 65536)) {
00270             ++ i;
00271             if (i == 5) {
00272                 TRACE_FATAL("Cann't find TC0 divisor!\n\r");
00273             }
00274         }
00275         div = divs[i];
00276         tcclks = i;
00277     }
00278     /* Configure TC for tioa output: 48M -> 48K*2 */
00279     TC_Configure(TC0,0, tcclks /*MCK/2*/
00280                         | TC_CMR_ACPC_SET
00281                         | TC_CMR_WAVE
00282                         | TC_CMR_ACPA_CLEAR
00283                         | TC_CMR_CPCTRG);
00284     /* 50% duty ,freq frequency*/
00285     ra = (((double)mck/div)/(freq*2) + 0.5 );
00286     rc = (((double)mck/div)/(freq)   + 0.99);
00287     TC0->TC_CHANNEL[0].TC_RA = (uint32_t)ra;
00288     TC0->TC_CHANNEL[0].TC_RC = (uint32_t)ra*2;
00289     bakRA = (uint32_t)ra;
00290 
00291     printf("-I- MCK %dKHz, Div %d(%x), RA: %d*.01(%x), RC: %d*.01(%x)\n\r",
00292             (int)mck/1000, (int)div, (unsigned int)div,
00293             (int)(ra*100), (int)ra, (int)(rc*100), (int)rc);
00294 
00295     /* Initialize DACC with HW as trigger */

00297                      1,
00298                      1,
00299                      0, /* DACC_MR_WORD_HALF */
00300                      0,
00301                      BOARD_MCK,
00302                      8,
00303                      CHANNEL_R,
00304                      1,
00305                      16);
00306     /* channel number is in the first 4 significant bits*/
00307     DACC->DACC_MR |= DACC_MR_TAG;
00308     /* disable PDC for DAC */
00309     DACC->DACC_PTCR = DACC_PTCR_TXTDIS;

00398 /**
00399  * Buffer translation for PCM format
00400  * \param pBuffer    Pointer for source and destination buffer
00401  * \param size       Buffer size in number of samples
00402  * \param nbChannels Channel numbers for source data
00403 */
00404 static void Pcm2DacTranslate(int16_t *pBuffer,
00405                              uint32_t size,
00406                              uint8_t nbChannels)
00407 {
00408     uint32_t i = 0;
00409     while(i <size){
00410         if(nbChannels > 1) {
00411             /* channle0-0,channel1-1*/
00412             pBuffer[i] = ( ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00413                            & MASK_12BITS )
00414                        | ( (uint16_t)(i%2) << 12);
00415         }
00416         else if (nbChannels == 1){
00417           /* one channel data stored continously */
00418           pBuffer[i] = ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00419                          & MASK_12BITS;
00420         }
00421         i++;
00422     }
00423 }

00440  * Handles interrupts coming from the DACC.
00441  */
00442 void DAC_IrqHandler(void)
00443 {
00444     Dacc *pDac = DACC;
00445     uint32_t sr = pDac->DACC_ISR;
00446 
00447     if (sr & DACC_ISR_TXBUFE) {
00448         /* End of transmission */
00449         AudioPlayEnable(0);
00450         DACC_DisableIt(pDac, DACC_IDR_TXBUFE | DACC_IDR_ENDTX);
00451         pDac->DACC_PTCR = DACC_PTCR_TXTDIS;
00452         isDacActive = 0;
00453         /* Reset playback index */
00454         inBufferIndex = outBufferIndex = 0;
00455     }
00456     else if (numBuffersToSend) {
00457         /* Check the number of available buffers */
00458         if (numBuffersToSend > DAC_DELAY) {
00459             //printf("%d+", numBuffersToSend);
00460             /* Speed up DAC freq (TC) */
00461             {   uint32_t newRA = bakRA - (numBuffersToSend-DAC_DELAY);
00462                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00463                     TC0->TC_CHANNEL[0].TC_RA =  newRA;
00464                     TC0->TC_CHANNEL[0].TC_RC =  newRA*2;
00465                 }
00466             }
00467         }
00468         else if (numBuffersToSend < DAC_DELAY) {
00469             //printf("%d-", numBuffersToSend);
00470             /* Slow down DAC (TC) */
00471             {   uint32_t newRA = bakRA + (DAC_DELAY-numBuffersToSend);
00472                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00473                     TC0->TC_CHANNEL[0].TC_RA = newRA;
00474                     TC0->TC_CHANNEL[0].TC_RC = newRA*2;
00475                 }
00476             }
00477         }
00478         else {
00479             /* Keep DAC (TC) */
00480             if (TC0->TC_CHANNEL[0].TC_RA != bakRA) {
00481                 TC0->TC_CHANNEL[0].TC_RA = bakRA;
00482                 TC0->TC_CHANNEL[0].TC_RC = bakRA*2;
00483             }
00484         }
00485         /* Load next buffer */
00486         DacPlayBuffer(buffers[outBufferIndex],
00487                       bufferSizes[outBufferIndex],
00488                       AUDDevice_NUMCHANNELS);
00489         outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00490         numBuffersToSend --;
00491     }
00492     else {
00493         /* Last frame remaining */
00494         DACC_DisableIt(pDac, DACC_IDR_ENDTX);
00495     }

00499  *  Invoked when a frame has been received.
00500  */
00501 static void FrameReceived(uint32_t unused,
00502                           uint8_t status,
00503                           uint32_t transferred,
00504                           uint32_t remaining)
00505 {
00506     Dacc *pDac = DACC;
00507 
00508     if (status == USBD_STATUS_SUCCESS) {
00509 
00510         bufferSizes[inBufferIndex] = transferred
00511                                         / AUDDevice_BYTESPERSAMPLE;
00512         inBufferIndex = (inBufferIndex + 1) % BUFFER_NUMBER;
00513         numBuffersToSend++;
00514 
00515         /* Start DAc transmission if necessary */
00516         if (!isDacActive) {
00517 
00518             dacDelay = DAC_DELAY;
00519             isDacActive = 1;
00520         }
00521         /* Wait until a few buffers have been received */
00522         else if (dacDelay > 0) {
00523 
00524             dacDelay--;
00525         }
00526         /* Start sending buffers */
00527         else if ((pDac->DACC_PTSR & DACC_PTSR_TXTEN) == 0) {
00528             AudioPlayEnable(1);
00529             DacPlayBuffer(buffers[outBufferIndex],
00530                           bufferSizes[outBufferIndex],
00531                           AUDDevice_NUMCHANNELS);
00532             outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00533             numBuffersToSend --;
00534             DacPlayBuffer(buffers[outBufferIndex],
00535                           bufferSizes[outBufferIndex],
00536                           AUDDevice_NUMCHANNELS);
00537             outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00538             numBuffersToSend --;
00539             DACC_EnableIt(pDac, DACC_IER_TXBUFE | DACC_IER_ENDTX);
00540             pDac->DACC_PTCR = DACC_PTCR_TXTEN;
00541         }
00542     }
00543     else if (status == USBD_STATUS_ABORTED) {
00544         /* Error , ABORT, add NULL buffer */
00545         bufferSizes[inBufferIndex] = 0;
00546         inBufferIndex = (inBufferIndex + 1) % BUFFER_NUMBER;
00547         numBuffersToSend++;
00548     }
00549     else {
00550         /* Packet is discarded */
00551     }
00552 
00553     /* Receive next packet */
00554     AUDDSpeakerDriver_Read(buffers[inBufferIndex],
00555                            AUDDevice_BYTESPERFRAME,
00556                            (TransferCallback) FrameReceived,
00557                            0); // No optional argument
00558 }
