00030 /** \addtogroup usart_module Working with USART
00031  * The USART driver provides the interface to configure and use the USART peripheral.\n
00032  *
00033  * The USART supports several kinds of comminication modes such as full-duplex asynchronous/
00034  * synchronous serial commnunication,RS485 with driver control signal,ISO7816,SPI and Test modes.
00035  *
00036  * To start a USART transfer with \ref AT91SAM3S_PDC "PDC" support, the user could follow these steps:
00037  * <ul>
00038  * <li> Configure USART with expected mode and baudrate(see \ref USART_Configure), which could be done by:
00039  * -# Resetting and disabling transmitter and receiver by setting US_CR(Control Register). </li>
00040  * -# Conifguring the USART in a specific mode by setting USART_MODE bits in US_MR(Mode Register) </li>
00041  * -# Setting baudrate which is different from mode to mode.
00042    </li>
00043  * <li> Enable transmitter or receiver respectively by set US_CR_TXEN or US_CR_RXEN in US_CR.</li>
00044  * <li> Read from or write to the peripheral with  \ref USART_ReadBuffer or \ref USART_WriteBuffer.
00045         These operations could be done by polling or interruption. </li>
00046  * <li> For polling, check the status bit US_CSR_ENDRX/US_CSR_RXBUFF (READ) or US_CSR_ENDTX/
00047         US_CSR_TXBUFE (WRITE).  </li>
00048  * <li> For interruption,"enable" the status bit through US_IER and
00049         realize the hanler with USARTx_IrqHandler according to IRQ vector
00050         table which is defined in board_cstartup_<toolchain>.c
00051         To enable the interruption of USART,it should be configured with priority and enabled first through
00052         NVIC .</li>
00053  * </ul>
00054  *
00055  * For more accurate information, please look at the USART section of the
00056  * Datasheet.
00057  *
00058  * Related files :\n
00059  * \ref usart.c\n
00060  * \ref usart.h\n
00061 */

00089 /**
00090  * \brief Configures an USART peripheral with the specified parameters.
00091  *
00092  *
00093  *  \param usart  Pointer to the USART peripheral to configure.
00094  *  \param mode  Desired value for the USART mode register (see the datasheet).
00095  *  \param baudrate  Baudrate at which the USART should operate (in Hz).
00096  *  \param masterClock  Frequency of the system master clock (in Hz).
00097  */
00098 void USART_Configure(Usart *usart,
00099                             uint32_t mode,
00100                             uint32_t baudrate,
00101                             uint32_t masterClock)
00102 {
00103     /* Reset and disable receiver & transmitter*/
00104     usart->US_CR = US_CR_RSTRX | US_CR_RSTTX
00105                    | US_CR_RXDIS | US_CR_TXDIS;
00106 
00107     /* Configure mode*/
00108     usart->US_MR = mode;
00109 
00110     /* Configure baudrate*/
00111     /* Asynchronous, no oversampling*/
00112     if ( ((mode & US_MR_SYNC) == 0) && ((mode & US_MR_OVER) == 0) )
00113     {
00114         usart->US_BRGR = (masterClock / baudrate) / 16;
00115     }
00116 
00117     if( ((mode & US_MR_USART_MODE_SPI_MASTER) == US_MR_USART_MODE_SPI_MASTER)
00118      || ((mode & US_MR_SYNC) == US_MR_SYNC))
00119     {
00120         if( (mode & US_MR_USCLKS_Msk) == US_MR_USCLKS_MCK)
00121         {
00122             usart->US_BRGR = masterClock / baudrate;
00123         }
00124         else
00125         {
00126             if ( (mode & US_MR_USCLKS_DIV) == US_MR_USCLKS_DIV)
00127             {
00128                 usart->US_BRGR = masterClock / baudrate / 8;
00129             }
00130         }
00131     }
00132     /* TODO other modes*/
00133 }
