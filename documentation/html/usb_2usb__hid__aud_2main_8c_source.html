00313 /**
00314  * Configure the TC0 and DACC for audio output.
00315  * \param sampleRate Audio sample rate.
00316  * \param nbChannels Number of audio channels.
00317  * \param mck        MCK frequence.
00318  */
00319 static void ConfigureAudioPlay(uint32_t sampleRate,
00320                                uint8_t nbChannels,
00321                                uint32_t mck)
00322 {
00323     uint32_t div = 2;
00324     uint32_t tcclks; // TC_CMR0_TCCLKS_TIMER_CLOCK1
00325     uint32_t freq = sampleRate * nbChannels;
00326     double   ra, rc;
00327 
00328     /* Enable TC0 Peripheral */
00329     PMC_EnablePeripheral(ID_TC0);
00330     {
00331         uint32_t divs[5] = {2, 8, 32, 128,BOARD_MCK / 32768};
00332         uint8_t i = 0;
00333         divs[4] = mck/32768;
00334         /* Minimize DIV & Maxmize RC for better waveform */
00335         while(freq < ((mck / divs[i]) / 65536)) {
00336             ++ i;
00337             if (i == 5) {
00338                 TRACE_FATAL("Cann't find TC0 divisor!\n\r");
00339             }
00340         }
00341         div = divs[i];
00342         tcclks = i;
00343     }
00344     /* Configure TC for tioa output: 48M -> 48K*2 */
00345     TC_Configure(TC0,0, tcclks /*MCK/2*/
00346                         | TC_CMR_ACPC_SET
00347                         | TC_CMR_WAVE
00348                         | TC_CMR_ACPA_CLEAR
00349                         | TC_CMR_CPCTRG);
00350     /* 50% duty ,freq frequency*/
00351     ra = (((double)mck/div)/(freq*2) + 0.5 );
00352     rc = (((double)mck/div)/(freq)   + 0.99);
00353     TC0->TC_CHANNEL[0].TC_RA = (uint32_t)ra;
00354     TC0->TC_CHANNEL[0].TC_RC = (uint32_t)ra*2;
00355     bakRA = (uint32_t)ra;
00356 
00357     printf("-I- MCK %dKHz, Div %d(%x), RA: %d*.01(%x), RC: %d*.01(%x)\n\r",
00358             (int)mck/1000, (int)div, (unsigned int)div,
00359             (int)(ra*100), (int)ra, (int)(rc*100), (int)rc);
00360 
00361     /* Initialize DACC with HW as trigger */
00362     DACC_Initialize( DACC, ID_DACC,
00363                      1,
00364                      1,
00365                      0, /* DACC_MR_WORD_HALF */
00366                      0,
00367                      BOARD_MCK,
00368                      8,
00369                      CHANNEL_R,
00370                      1,
00371                      16);
00372     /* channel number is in the first 4 significant bits*/
00373     DACC->DACC_MR |= DACC_MR_TAG;
00374     /* disable PDC for DAC */
00375     DACC->DACC_PTCR = DACC_PTCR_TXTDIS;
00376     /* Mute */
00377     AudioPlayEnable(0);
00378     /* Enable DACC ISR */
00379     NVIC_EnableIRQ(DACC_IRQn);

00438 /**
00439  * Buffer translation for PCM format
00440  * \param pBuffer    Pointer for source and destination buffer
00441  * \param size       Buffer size in number of samples
00442  * \param nbChannels Channel numbers for source data
00443 */
00444 static void Pcm2DacTranslate(int16_t *pBuffer,
00445                              uint32_t size,
00446                              uint8_t nbChannels)
00447 {
00448     uint32_t i = 0;
00449     while(i <size){
00450         if(nbChannels > 1) {
00451             /* channle0-0,channel1-1*/
00452             pBuffer[i] = ( ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00453                            & MASK_12BITS )
00454                        | ( (uint16_t)(i%2) << 12);
00455         }
00456         else if (nbChannels == 1){
00457           /* one channel data stored continously */
00458           pBuffer[i] = ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00459                          & MASK_12BITS;
00460         }
00461         i++;
00462     }
00463 }

00479 /**
00480  * Handles interrupts coming from the DACC.
00481  */
00482 void DAC_IrqHandler(void)
00483 {
00484     Dacc *pDac = DACC;
00485     uint32_t sr = pDac->DACC_ISR;
00486 
00487     if (sr & DACC_ISR_TXBUFE) {
00488         /* End of transmission */
00489         AudioPlayEnable(0);
00490         DACC_DisableIt(pDac, DACC_IDR_TXBUFE | DACC_IDR_ENDTX);
00491         pDac->DACC_PTCR = DACC_PTCR_TXTDIS;
00492         isDacActive = 0;
00493         /* Reset playback index */
00494         inBufferIndex = outBufferIndex = 0;
00495     }
00496     else if (numBuffersToSend) {
00497         /* Check the number of available buffers */
00498         if (numBuffersToSend > DAC_DELAY) {
00499             //printf("%d+", numBuffersToSend);
00500             /* Speed up DAC freq (TC) */
00501             {   uint32_t newRA = bakRA - (numBuffersToSend-DAC_DELAY);
00502                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00503                     TC0->TC_CHANNEL[0].TC_RA =  newRA;
00504                     TC0->TC_CHANNEL[0].TC_RC =  newRA*2;
00505                 }
00506             }
00507         }
00508         else if (numBuffersToSend < DAC_DELAY) {
00509             //printf("%d-", numBuffersToSend);
00510             /* Slow down DAC (TC) */
00511             {   uint32_t newRA = bakRA + (DAC_DELAY-numBuffersToSend);
00512                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00513                     TC0->TC_CHANNEL[0].TC_RA = newRA;
00514                     TC0->TC_CHANNEL[0].TC_RC = newRA*2;
00515                 }
00516             }
00517         }
00518         else {
00519             /* Keep DAC (TC) */
00520             if (TC0->TC_CHANNEL[0].TC_RA != bakRA) {
00521                 TC0->TC_CHANNEL[0].TC_RA = bakRA;
00522                 TC0->TC_CHANNEL[0].TC_RC = bakRA*2;
00523             }
00524         }
00525         /* Load next buffer */
00526         DacPlayBuffer(buffers[outBufferIndex],
00527                       bufferSizes[outBufferIndex],
00528                       AUDDevice_NUMCHANNELS);
00529         outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00530         numBuffersToSend --;
00531     }
00532     else {
00533         /* Last frame remaining */
00534         DACC_DisableIt(pDac, DACC_IDR_ENDTX);
00535     }

00543                           uint32_t transferred,
00544                           uint32_t remaining)
00545 {
00546     Dacc *pDac = DACC;
00547 
00548     if (status == USBD_STATUS_SUCCESS) {
00549

00551                                         / AUDDevice_BYTESPERSAMPLE;

00557

00560         }
00561         /* Wait until a few buffers have been received */
00562         else if (dacDelay > 0) {
00563 
00564             dacDelay--;
00565         }
00566         /* Start sending buffers */
00567         else if ((pDac->DACC_PTSR & DACC_PTSR_TXTEN) == 0) {
00568             AudioPlayEnable(1);
00569             DacPlayBuffer(buffers[outBufferIndex],
00570                           bufferSizes[outBufferIndex],
00571                           AUDDevice_NUMCHANNELS);
00572             outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00573             numBuffersToSend --;
00574             DacPlayBuffer(buffers[outBufferIndex],
00575                           bufferSizes[outBufferIndex],
00576                           AUDDevice_NUMCHANNELS);

00578             numBuffersToSend --;
00579             DACC_EnableIt(pDac, DACC_IER_TXBUFE | DACC_IER_ENDTX);
00580             pDac->DACC_PTCR = DACC_PTCR_TXTEN;

00589     else {

00591     }
00592 
00593     /* Receive next packet */

00595                       AUDDevice_BYTESPERFRAME,
00596                       (TransferCallback) FrameReceived,
00597                       0); // No optional argument
