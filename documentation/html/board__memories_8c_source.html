00104 /**
00105  * \brief An accurate one-to-one comparison is necessary between PSRAM and SMC waveforms for
00106  *   a complete SMC configuration.
00107  *  \note The system is running at 48 MHz for the EBI Bus.
00108  *        Please refer to the "AC Characteristics" section of the customer product datasheet.
00109  */
00110 extern void BOARD_ConfigurePSRAM( Smc* pSmc )
00111 {
00112     uint32_t dwTmp ;
00113 
00114     /* Enable peripheral clock */
00115     PMC_EnablePeripheral( ID_SMC ) ;
00116 
00117     /* Configure SMC, NCS1 is assigned to a external PSRAM */
00118     /**
00119      * PSRAM IS66WV51216BLL
00120      * 55 ns Access time
00121      * tdoe = 25 ns max
00122      * SMC1 (timing SAM3S read mode SMC) = 21 ns of setup
00123      * 21 + 55 = 76 ns => at least 5 cycles at 64 MHz
00124      * Write pulse width minimum = 45 ns (PSRAM)
00125      */
00126     pSmc->SMC_CS_NUMBER[1].SMC_SETUP = SMC_SETUP_NWE_SETUP( 1 )
00127                                      | SMC_SETUP_NCS_WR_SETUP( 0 )
00128                                      | SMC_SETUP_NRD_SETUP( 2 )
00129                                      | SMC_SETUP_NCS_RD_SETUP( 0 ) ;
00130 
00131     pSmc->SMC_CS_NUMBER[1].SMC_PULSE = SMC_PULSE_NWE_PULSE( 3 )
00132                                      | SMC_PULSE_NCS_WR_PULSE( 4 )
00133                                      | SMC_PULSE_NRD_PULSE( 3 )
00134                                      | SMC_PULSE_NCS_RD_PULSE( 5 ) ;
00135 
00136     /* NWE_CYCLE:     The total duration of the write cycle.
00137                       NWE_CYCLE = NWE_SETUP + NWE_PULSE + NWE_HOLD
00138                       = NCS_WR_SETUP + NCS_WR_PULSE + NCS_WR_HOLD
00139                       (tWC) Write Cycle Time min. 70ns
00140        NRD_CYCLE:     The total duration of the read cycle.
00141                       NRD_CYCLE = NRD_SETUP + NRD_PULSE + NRD_HOLD
00142                       = NCS_RD_SETUP + NCS_RD_PULSE + NCS_RD_HOLD
00143                       (tRC) Read Cycle Time min. 70ns. */
00144     pSmc->SMC_CS_NUMBER[1].SMC_CYCLE = SMC_CYCLE_NWE_CYCLE( 4 )
00145                                      | SMC_CYCLE_NRD_CYCLE( 5 ) ;
00146 
00147     dwTmp = SMC->SMC_CS_NUMBER[0].SMC_MODE & (uint32_t)(~(SMC_MODE_DBW_Msk)) ;
00148     pSmc->SMC_CS_NUMBER[1].SMC_MODE  = dwTmp
00149                                      | SMC_MODE_READ_MODE
00150                                      | SMC_MODE_WRITE_MODE
00151                                      | SMC_MODE_DBW_8_BIT ;
00152 }
