00030 /** \addtogroup dacc_module Working with DACC
00031  * The DACC driver provides the interface to configure and use the DACC peripheral.\n
00032  *
00033  * The DACC(Digital-to-Analog Converter Controller) converts digital code to analog output.
00034  * The data to be converted are sent in a common register for all channels. It offers up to 2
00035  * analog outputs.The output voltage ranges from (1/6)ADVREF to (5/6)ADVREF.
00036  *
00037  * To Enable a DACC conversion,the user has to follow these few steps:
00038  * <ul>
00039  * <li> Select an appropriate reference voltage on ADVREF   </li>
00040  * <li> Configure the DACC according to its requirements and special needs,which could be
00041         broken down into several parts:
00042  * -#   Enable DACC in free running mode by clearing TRGEN in DACC_MR;
00043  * -#   Configure Startup Time and Refresh Period through setting STARTUP and REFRESH fields
00044  *      in DACC_MR; The refresh mechanism is used to protect the output analog value from
00045  *      decreasing.
00046  * -#   Enable channels and write digital code to DACC_CDR,in free running mode, the conversion
00047  *      is started right after at least one channel is enabled and data is written .
00048    </li>
00049  * </ul>
00050  *
00051  * For more accurate information, please look at the DACC section of the
00052  * Datasheet.
00053  *
00054  * Related files :\n
00055  * \ref DACC.c\n
00056  * \ref DACC.h\n
00057 */

00079 /**
00080   * \brief Initialize the DACC controller
00081   * \param pDACC Pointer to an DACC instance.
00082   * \param idDACC identifier of DAC peripheral
00083   * \param trgEn trigger mode, free running mode or external Hardware trigger
00084   * \param word transfer size,word or half word
00085   * \param trgSel hardware trigger selection
00086   * \param sleepMode sleep mode selection
00087   * \param mck value of MCK in Hz
00088   * \param refresh refresh period
00089   * \param user_sel user channel selection ,0 or 1
00090   * \param tag_mode tag for channel number
00091   * \param startup value of the start up time (in DACCClock) (see datasheet)
00092 */
00093 extern void DACC_Initialize( Dacc* pDACC,
00094                              uint8_t idDACC,
00095                              uint8_t trgEn,
00096                              uint8_t trgSel,
00097                              uint8_t word,
00098                              uint8_t sleepMode,
00099                              uint32_t mck,
00100                              uint8_t refresh,    /* refresh period */
00101                              uint8_t user_sel,   /* user channel selection */
00102                              uint32_t tag_mode,  /* using tag for channel number */
00103                              uint32_t startup
00104                             )
00105 {
00106     assert( 1024*refresh*1000/(mck>>1) < 20 ) ;
00107 
00108     /* Enable peripheral clock*/
00109     PMC->PMC_PCER0 = 1 << idDACC;
00110 
00111     /*  Reset the controller */
00112     DACC_SoftReset(pDACC);
00113 
00114     /*  Write to the MR register */
00115     DACC_CfgModeReg( pDACC,
00116           ( (trgEn<<0) & DACC_MR_TRGEN)
00117         |   DACC_MR_TRGSEL(trgSel)
00118         | ( (word<<4) & DACC_MR_WORD)
00119         | ( (sleepMode<<5) & DACC_MR_SLEEP)
00120         |   DACC_MR_REFRESH(refresh)
00121         | ( (user_sel<<DACC_MR_USER_SEL_Pos)& DACC_MR_USER_SEL_Msk)
00122         | ( (tag_mode<<20) &  DACC_MR_TAG)
00123         | ( (startup<<DACC_MR_STARTUP_Pos) & DACC_MR_STARTUP_Msk));
00124 }
