00169 /**
00170  * Configure the  MCI CLKDIV in the MCI_MR register. The max. for MCI clock is
00171  * MCK/2 and corresponds to CLKDIV = 0
00172  * \param pMci  Pointer to the low level MCI driver.
00173  * \param mciSpeed  MCI clock speed in Hz, 0 will not change current speed.
00174  * \param mck       MCK to generate MCI Clock, in Hz
00175  * \return The actual speed used, 0 for fail.
00176  */
00177 uint32_t MCI_SetSpeed( Mcid* pMci, uint32_t mciSpeed, uint32_t mck )
00178 {
00179     Hsmci *pMciHw = pMci->pMciHw;
00180     uint32_t mciMr;
00181     uint32_t clkdiv;
00182     uint8_t  mciDis;
00183 
00184     assert(pMci);
00185     assert(pMciHw);
00186 
00187     PMC_EnablePeripheral(pMci->mciId);
00188     mciDis = PMC_IsPeriphEnabled(pMci->mciId);
00189 
00190     mciMr = pMciHw->HSMCI_MR & (~(uint32_t)HSMCI_MR_CLKDIV_Msk);
00191     /* Multimedia Card Interface clock (MCCK or MCI_CK) is Master Clock (MCK)
00192      * divided by (2*(CLKDIV+1))
00193      * mciSpeed = MCK / (2*(CLKDIV+1)) */
00194     if (mciSpeed > 0)
00195     {
00196         clkdiv = (mck / 2 / mciSpeed);
00197         /* Speed should not bigger than expired one */
00198         if (mciSpeed < mck/2/clkdiv)
00199         {
00200             clkdiv ++;
00201         }
00202         
00203         if ( clkdiv > 0 )
00204         {
00205             clkdiv -= 1;
00206         }
00207         assert( (clkdiv & 0xFFFFFF00) == 0 ) ; /* "mciSpeed too small" */
00208     }
00209     else
00210     {
00211         clkdiv = 0 ;
00212     }
00213 
00214     /* Actual MCI speed */
00215     mciSpeed = mck / 2 / (clkdiv + 1);
00216     /* Modify MR */
00217     pMciHw->HSMCI_MR = mciMr | clkdiv;
00218     if ( mciDis )
00219     {
00220         PMC_DisablePeripheral( pMci->mciId ) ;
00221     }
00222 
00223     return (mciSpeed);
00224 }
