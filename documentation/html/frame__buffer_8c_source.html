00228  * Pixel color is set by the LCD_SetColor() function.

00230  * individual pixels in order to limit the number of SPI interrupts.
00231  * \param dwX1      X-coordinate of upper-left corner on LCD.
00232  * \param dwY1      Y-coordinate of upper-left corner on LCD.
00233  * \param dwX2      X-coordinate of lower-right corner on LCD.
00234  * \param dwY2      Y-coordinate of lower-right corner on LCD.
00235  *
00236  * \return 0 if operation is successfull
00237  */
00238 extern uint32_t FB_DrawFilledRectangle( uint32_t dwX1, uint32_t dwY1, uint32_t dwX2, uint32_t dwY2 )
00239 {
00240     LcdColor_t *pFbBuffer, *pDestFbBuffer;
00241     uint32_t dwY, blocks, bytes;
00242 
00243     /* Swap coordinates if necessary */
00244     CheckBoxCoordinates(&dwX1, &dwY1, &dwX2, &dwY2);
00245 
00246     blocks  = ((dwX2 - dwX1 + 1) / FB_COLOR_CACHE_SIZE);
00247     bytes   = ((dwX2 - dwX1 + 1) % FB_COLOR_CACHE_SIZE);
00248 
00249     /* Each row is sent by block to benefit from memcpy optimizations */
00250     pFbBuffer = &(gpBuffer[dwY1 * gucWidth + dwX1]);
00251     for (dwY = dwY1; dwY <= dwY2; ++dwY) {
00252         pDestFbBuffer = pFbBuffer;
00253         while (blocks--) {
00254             memcpy(pDestFbBuffer, gFbPixelCache, FB_COLOR_CACHE_SIZE * sizeof(LcdColor_t));
00255             pDestFbBuffer += FB_COLOR_CACHE_SIZE;
00256 
00257         }
00258        memcpy(pDestFbBuffer, gFbPixelCache, bytes * sizeof(LcdColor_t));
00259        pFbBuffer += gucWidth;
00260     }
00261 
00262     return 0;
