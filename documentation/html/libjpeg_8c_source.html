00073 /**
00074   * \brief initial mcu rows with yuv420 data
00075   * \param image pointer to mcu row
00076   * \param src pointer of source with yuv422 format
00077   * \param y_pos the start position of the mcu rows in a whole image
00078   * \param width width of the image
00079   * \param rows total number of concerned rows in the mcu
00080   */
00081 static void _init_sarray_imcu_row(JSAMPIMAGE *image,JSAMPLE *src,uint32_t y_pos,uint32_t width,uint32_t rows)
00082 {
00083     JSAMPLE *r0;
00084     JSAMPLE *r1;
00085     JSAMPARRAY ay = (*image)[0];
00086     JSAMPARRAY au = (*image)[1];
00087     JSAMPARRAY av = (*image)[2];
00088     JSAMPROW ry0,ru,rv,ry1;    
00089     
00090     int i,j;
00091     for( i = 0; i < rows/2 ;i++)
00092     {
00093         r0 = & src[ (i * 2 + y_pos) * width * 2];
00094         r1 = & src[ (i * 2 + y_pos + 1) * width * 2];
00095         
00096         ry0 = ay[i*2];
00097         ry1 = ay[i*2+1];
00098         
00099         ru = au[i];
00100         rv = av[i];
00101         
00102         for(j = 0; j < width /2; j++)
00103         {
00104             ry0[j * 2] = r0[j*4];
00105             ry1[j * 2] = r1[j*4];
00106             ry0[j * 2 + 1] = r0[j*4+2];
00107             ry1[j * 2 +1] = r1[j*4+2];
00108             
00109             ru[j] = (r0[ j * 4 + 1] + r1[j*4+1])/2;
00110             rv[j] = (r0[ j * 4 + 3] + r1[j*4+3])/2;
00111             
00112         }
00113         
00114         
00115     }
00116 }
