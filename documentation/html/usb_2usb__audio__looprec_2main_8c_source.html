00033  *  \section Purpose
00034  *
00035  *  The USB Audio Speaker & Loopback-Recorder Example will help you to get
00036  *  familiar with the USB Device Port(UDP) and DACC on SAM microcontrollers.

00039  *  class.

00041  *  You can find following information depends on your needs:
00042  *  - Sample usage of USB Audio Device Class driver and DACC driver.
00043  *  - USB Audio Class driver development based on the USB Framework.
00044  *  - USB enumerate sequence, the standard and class-specific descriptors and

00051  *  - dacc: DACC interface driver

00054  *         - \ref usbd_api

00057  *  \section Requirements
00058  *
00059  *  This package can be used with SAM3S evaluation kits that have both
00060  *  UDP and DACC.
00061  *
00062  *  \section Description
00063  *
00064  *  When an EK running this program connected to a host (PC for example), with

00066  *  can play sound through host software. The %audio stream from the host is
00067  *  then sent to the EK, and eventually sent to %audio DAC connected to the
00068  *  amplifier. At the same time, the %audio stream received is also sent
00069  *  back to host from EK for recording.

00260 /**
00261  * Configure the TC0 and DACC for audio output.
00262  * \param sampleRate Audio sample rate.
00263  * \param nbChannels Number of audio channels.
00264  * \param mck        MCK frequence.
00265  */

00267                                uint8_t nbChannels,
00268                                uint32_t mck)
00269 {
00270     uint32_t div = 2;
00271     uint32_t tcclks = TC_CMR_TCCLKS_TIMER_CLOCK1;
00272     uint32_t freq = sampleRate * nbChannels;
00273     double   ra, rc;
00274 
00275     /* Enable TC0 Peripheral */
00276     PMC_EnablePeripheral(ID_TC0);

00278         uint32_t divs[5] = {2, 8, 32, 128,BOARD_MCK / 32768};
00279         uint8_t i = 0;
00280         divs[4] = mck/32768;
00281         /* Minimize DIV & Maxmize RC for better waveform */
00282         while(freq < ((mck / divs[i]) / 65536)) {
00283             ++ i;
00284             if (i == 5) {
00285                 TRACE_FATAL("Cann't find TC0 divisor!\n\r");
00286             }
00287         }
00288         div = divs[i];
00289         tcclks = i;
00290     }
00291     /* Configure TC for tioa output: 48M -> 48K*2 */
00292     TC_Configure(TC0,0, tcclks /*MCK/2*/
00293                         | TC_CMR_ACPC_SET
00294                         | TC_CMR_WAVE
00295                         | TC_CMR_ACPA_CLEAR
00296                         | TC_CMR_CPCTRG);
00297     /* 50% duty ,freq frequency*/
00298     ra = (((double)mck/div)/(freq*2) + 0.5 );
00299     rc = (((double)mck/div)/(freq)   + 0.99);
00300     TC0->TC_CHANNEL[0].TC_RA = (uint32_t)ra;
00301     TC0->TC_CHANNEL[0].TC_RC = (uint32_t)ra*2;
00302     defaultRA = (uint32_t)ra;
00303 
00304     printf("-I- MCK %dKHz, Div %d(%x), RA: %d*.01(%x), RC: %d*.01(%x)\n\r",
00305             (int)mck/1000, (int)div, (unsigned int)div,
00306             (int)(ra*100), (int)ra, (int)(rc*100), (int)rc);
00307 
00308     /* Initialize DACC with HW as trigger */

00310                      1,
00311                      1,
00312                      0, /* DACC_MR_WORD_HALF */
00313                      0,
00314                      BOARD_MCK,
00315                      8,
00316                      CHANNEL_R,
00317                      1,
00318                      16);
00319     /* channel number is in the first 4 significant bits*/
00320     DACC->DACC_MR |= DACC_MR_TAG;
00321     /* disable PDC for DAC */

00333 /**
00334  * Buffer translation for PCM format
00335  * \param pBuffer    Pointer for source and destination buffer
00336  * \param size       Buffer size in number of samples
00337  * \param nbChannels Channel numbers for source data
00338 */
00339 static void Pcm2DacTranslate(int16_t *pBuffer,
00340                              uint32_t size,
00341                              uint8_t nbChannels)
00342 {
00343     uint32_t i = 0;
00344     while(i <size){
00345         if(nbChannels > 1) {
00346             /* channle0-0,channel1-1*/
00347             pBuffer[i] = ( ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00348                            & MASK_12BITS )
00349                        | ( (uint16_t)(i%2) << 12);
00350         }
00351         else if (nbChannels == 1){
00352           /* one channel data stored continously */
00353           pBuffer[i] = ( (pBuffer[i] + MID_VALUE_16BITS) >> 4 )
00354                          & MASK_12BITS;
00355         }
00356         i++;
00357     }
00358 }

00374 /**
00375  * Handles interrupts coming from the DACC.
00376  */
00377 void DAC_IrqHandler(void)
00378 {
00379     Dacc *pDac = DACC;
00380     uint32_t sr = pDac->DACC_ISR;
00381 
00382     if (sr & DACC_ISR_TXBUFE) {
00383         /* End of transmission */
00384         AudioPlayEnable(0);
00385         DACC_DisableIt(pDac, DACC_IDR_TXBUFE | DACC_IDR_ENDTX);
00386         pDac->DACC_PTCR = DACC_PTCR_TXTDIS;
00387         isDacActive = 0;
00388     }
00389     else if (numBuffersToSend) {
00390         /* Check the number of available buffers */
00391         if (numBuffersToSend > DAC_DELAY) {
00392             //printf("%d+", numBuffersToSend);
00393             /* Speed up DAC freq (TC) */
00394             {   uint32_t newRA = defaultRA - (numBuffersToSend-DAC_DELAY);
00395                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00396                     TC0->TC_CHANNEL[0].TC_RA =  newRA;
00397                     TC0->TC_CHANNEL[0].TC_RC =  newRA*2;
00398                 }
00399             }
00400         }
00401         else if (numBuffersToSend < DAC_DELAY) {
00402             //printf("%d-", numBuffersToSend);
00403             /* Slow down DAC (TC) */
00404             {   uint32_t newRA = defaultRA + (DAC_DELAY-numBuffersToSend);
00405                 if (TC0->TC_CHANNEL[0].TC_RA != newRA) {
00406                     TC0->TC_CHANNEL[0].TC_RA = newRA;
00407                     TC0->TC_CHANNEL[0].TC_RC = newRA*2;
00408                 }
00409             }
00410         }
00411         else {
00412             /* Keep DAC (TC) */
00413             if (TC0->TC_CHANNEL[0].TC_RA != defaultRA) {
00414                 TC0->TC_CHANNEL[0].TC_RA = defaultRA;
00415                 TC0->TC_CHANNEL[0].TC_RC = defaultRA*2;
00416             }
00417         }
00418         /* Load next buffer */
00419         DacPlayBuffer(buffers[outBufferIndex],
00420                       bufferSizes[outBufferIndex],
00421                       AUDDevice_NUMCHANNELS);
00422         outBufferIndex = (outBufferIndex + 1) % BUFFER_NUMBER;
00423         numBuffersToSend --;
00424     }
00425     else {
00426         /* Last frame remaining */
00427         DACC_DisableIt(pDac, DACC_IDR_ENDTX);
00428     }
00429 }
