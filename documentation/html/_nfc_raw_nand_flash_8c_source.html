00105 /**
00106  * \brief Translates the given column and row address into first and other (1-4) address
00107  *  cycles. The resulting values are stored in the provided variables if they are not null.
00108  * \param columnAddress  Column address to translate.
00109  * \param rowAddress  Row address to translate.
00110  * \param pAddressCycle0  First address cycle.
00111  * \param pAddressCycle1234 four address cycles.
00112  */
00113 void NFC_TranslateAddress(
00114     const struct RawNandFlash *raw,
00115     unsigned short columnAddress,
00116     unsigned int rowAddress,
00117     unsigned int *pAddressCycle0,
00118     unsigned int *pAddressCycle1234,
00119     unsigned char useFiveAddress)
00120 {
00121     unsigned short pageDataSize = NandFlashModel_GetPageDataSize(MODEL(raw));
00122     unsigned int numPages = NandFlashModel_GetDeviceSizeInPages(MODEL(raw));
00123     unsigned char numAddressCycles = 0;
00124     unsigned int addressCycle0 = 0;
00125     unsigned int addressCycle1234 = 0;
00126 
00127     /* Check the data bus width of the NandFlash */
00128     if (NandFlashModel_GetDataBusWidth(MODEL(raw)) == 16) {
00129         /* Div 2 is because we address in word and not in byte */
00130         columnAddress >>= 1;
00131     }
00132     /* When 5 address cycle are used.*/
00133     if (useFiveAddress) {
00134 
00135         while (pageDataSize > 0) {
00136             if (numAddressCycles == 0) {
00137                 addressCycle0 = (columnAddress & 0xFF);
00138             }
00139             else {
00140                 addressCycle1234 |= (columnAddress & 0xFF) << ((numAddressCycles - 1) * 8);
00141             }
00142             pageDataSize >>= 8;
00143             columnAddress >>= 8;
00144             numAddressCycles ++;
00145         }
00146         while (numPages > 0) {
00147             if (numAddressCycles == 0) {
00148                 addressCycle0 = (rowAddress & 0xFF);
00149             }
00150             else {
00151                 addressCycle1234 |= (rowAddress & 0xFF) << ((numAddressCycles - 1) * 8);
00152             }
00153             numPages >>= 8;
00154             rowAddress >>= 8;
00155             numAddressCycles ++;
00156         }
00157     }
00158     /* When less than 5 address cycle are used.*/
00159     else {
00160         while (numPages > 0) {
00161             addressCycle1234 |= (rowAddress & 0xFF) << ((numAddressCycles) * 8);
00162             numPages >>= 8;
00163             rowAddress >>= 8;
00164             numAddressCycles ++;
00165         }
00166     }
00167     /* Store values*/
00168     if (pAddressCycle0) {
00169         *pAddressCycle0 = addressCycle0;
00170     }
00171     if (pAddressCycle1234) {
00172         *pAddressCycle1234 = addressCycle1234;
00173     }
00174     return;
00175 }
