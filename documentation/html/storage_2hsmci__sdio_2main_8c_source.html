00226 /**
00227  * \brief Get 32-bit number input (Dec or Hex).
00228  * Before first character input format can be changed once by
00229  * 'x' to hex.
00230  * \param nbChar Number of character to wait.
00231  * \param pNum   Pointer to uint32_t for input result.
00232  * \return 0 if valid data inputed.
00233  */
00234 static uint8_t GetU32Input(uint8_t nbChar, uint32_t *pU32)
00235 {
00236     uint8_t key, isHex = 0;
00237     uint32_t  i;
00238     uint32_t  result = 0;
00239     for (i = 0; i < nbChar;) {
00240         key = UART_GetChar();
00241         /* User cancel input */
00242         if (key == 27) {
00243             printf(" Canceled\n\r");
00244             return key;
00245         }
00246         /* Short input */
00247         if (key == '\r') break;
00248         if (key == 'x' && i == 0) {
00249             if (isHex == 0) {
00250                 isHex = 1;
00251                 UART_PutChar(key);
00252             }
00253             continue;
00254         }
00255         if (key > '9' || key < '0') {
00256             if (isHex) {
00257                 if (key < 'a' || key > 'z') {
00258                     continue;
00259                 }
00260             }
00261             else {
00262                 continue;
00263             }
00264         }
00265         UART_PutChar(key);
00266         if (isHex) {
00267             if (key >= 'a')
00268                 result = result * 16 + (key - 'a' + 10);
00269             else
00270                 result = result * 16 + (key - '0');
00271         }
00272         else {
00273             result = result * 10 + (key - '0');
00274         }
00275         i ++;
00276     }
00277     if (pU32) *pU32 = result;
00278     return 0;
00279 }
