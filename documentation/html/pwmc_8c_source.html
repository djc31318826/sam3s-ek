00030 /** \addtogroup pwm_module Working with PWM
00031  * The PWM driver provides the interface to configure and use the PWM
00032  * peripheral.
00033  *
00034  * The PWM macrocell controls square output waveforms of 4 channels.
00035  * Characteristics of output waveforms such as period, duty-cycle,
00036  * dead-time can be configured.\n
00037  * Some of PWM channels can be linked together as synchronous channel and
00038  * duty-cycle of synchronous channels can be updated by PDC automaticly.
00039  *
00040  * Before enabling the channels, they must have been configured first.
00041  * The main settings include:
00042  * <ul>
00043  * <li>Configuration of the clock generator.</li>
00044  * <li>Selection of the clock for each channel.</li>
00045  * <li>Configuration of output waveform characteristics, such as period, duty-cycle etc.</li>
00046  * <li>Configuration for synchronous channels if needed.</li>
00047  *    - Selection of the synchronous channels.
00048  *    - Selection of the moment when the WRDY flag and the corresponding PDC
00049  *      transfer request are set (PTRM and PTRCS in the PWM_SCM register).
00050  *    - Configuration of the update mode (UPDM in the PWM_SCM register).
00051  *    - Configuration of the update period (UPR in the PWM_SCUP register).
00052  * </ul>
00053  *
00054  * After the channels is enabled, the user must use respective update registers
00055  * to change the wave characteristics to prevent unexpected output waveform.
00056  * i.e. PWM_CDTYUPDx register should be used if user want to change duty-cycle
00057  * when the channel is enabled.
00058  *
00059  * For more accurate information, please look at the PWM section of the
00060  * Datasheet.
00061  *
00062  * Related files :\n
00063  * \ref pwmc.c\n
00064  * \ref pwmc.h.\n
00065  */

00089 /**
00090  * \brief Finds a prescaler/divisor couple to generate the desired frequency
00091  * from MCK.
00092  *
00093  * Returns the value to enter in PWM_CLK or 0 if the configuration cannot be
00094  * met.
00095  *
00096  * \param frequency  Desired frequency in Hz.
00097  * \param mck  Master clock frequency in Hz.
00098  */
00099 static uint16_t FindClockConfiguration(
00100     uint32_t frequency,
00101     uint32_t mck)
00102 {
00103     uint32_t divisors[11] = {1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024};
00104     uint8_t divisor = 0;
00105     uint32_t prescaler;
00106 
00107     assert(frequency < mck);
00108 
00109     /* Find prescaler and divisor values */
00110     prescaler = (mck / divisors[divisor]) / frequency;
00111     while ((prescaler > 255) && (divisor < 11)) {
00112 
00113         divisor++;
00114         prescaler = (mck / divisors[divisor]) / frequency;
00115     }
00116 
00117     /* Return result */
00118     if ( divisor < 11 )
00119     {
00120         TRACE_DEBUG( "Found divisor=%u and prescaler=%u for freq=%uHz\n\r", divisors[divisor], prescaler, frequency ) ;
00121 
00122         return prescaler | (divisor << 8) ;
00123     }
00124     else
00125     {
00126         return 0 ;
00127     }
00128 }

00225 {
00226     uint32_t mode = 0;
00227     uint32_t result;
00228
