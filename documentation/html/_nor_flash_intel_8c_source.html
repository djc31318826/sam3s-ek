00192  * \brief It implement a program word command. Returns 0 if the operation was
00193  * successful; otherwise returns an error code.
00194  *
00195  * \param pNorFlashInfo  Pointer to an struct NorFlashInfo instance.
00196  * \param address Start address offset to be wrote.
00197  * \param data word to be written.
00198  */
00199 static uint8_t intel_Program( NorFlashInfo *pNorFlashInfo, uint32_t address, uint32_t data )
00200 {
00201     uint32_t status;
00202     /* uint32_t datain; */
00203     volatile uint32_t busAddress;
00204     uint8_t done = 0;
00205     uint8_t busWidth;
00206 
00207     busWidth = NorFlash_GetDataBusWidth(pNorFlashInfo);
00208      /* Issue Read Array Command - just in case that the flash is not in Read Array mode */
00209     intel_Reset(pNorFlashInfo, address);
00210 
00211     busAddress = NorFlash_GetAddressInChip(pNorFlashInfo, address);
00212     /* Word programming operations are initiated by writing the Word Program Setup command to the device. */
00213     WriteCommand(busWidth, busAddress, INTEL_CMD_PROGRAM_WORD);
00214     /* This is followed by a second write to the device with the address and data to be programmed. */
00215     WriteRawData(busWidth, busAddress, (uint8_t*)&data);
00216 
00217     /* Status register polling */
00218     do
00219     {
00220         status = intel_ReadStatus(pNorFlashInfo,address);
00221         /* Check if the device is ready. */
00222         if ((status & INTEL_STATUS_DWS) == INTEL_STATUS_DWS )
00223         {
00224             /* check if VPP within acceptable limits during program or erase operation. */
00225             if ((status & INTEL_STATUS_VPPS) == INTEL_STATUS_VPPS )
00226             {
00227                 return NorCommon_ERROR_CANNOTWRITE;
00228             }
00229 
00230             /* Check if the erase block operation is completed. */
00231             if ((status & INTEL_STATUS_PS) == INTEL_STATUS_PS )
00232             {
00233                 return NorCommon_ERROR_CANNOTWRITE;
00234             }
00235             /* check if Block locked during program or erase, operation aborted. */
00236             else
00237             {
00238                 if ((status & INTEL_STATUS_BLS) == INTEL_STATUS_BLS )
00239                 {
00240                     return NorCommon_ERROR_CANNOTWRITE;
00241                 }
00242                 else
00243                 {
00244                     done = 1;
00245                 }
00246             }
00247         }
00248     } while ( !done ) ;
00249
