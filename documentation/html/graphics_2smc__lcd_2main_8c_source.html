00139 /**
00140  * \brief Make a test Image with special pattern.
00141  *
00142  * \param pBuffer Image buffer.
00143  *
00144  * \note The image is a raw data of RGB 24-bit format.
00145  */
00146 static void MakeTestImage( uint8_t *pBuffer )
00147 {
00148     uint32_t v_max  = IMAGE_HEIGHT;
00149     uint32_t h_max  = IMAGE_WIDTH;
00150     uint32_t v_step = (v_max + N_BLK_VERT - 1) / N_BLK_VERT;
00151     uint32_t h_step = (h_max + N_BLK_HOR  - 1) / N_BLK_HOR;
00152     uint32_t v, h;
00153     uint8_t  *pImage = (uint8_t *)pBuffer;
00154     uint8_t ix ;
00155     uint8_t iy ;
00156 
00157     for ( v = 0; v < v_max; v++ )
00158     {
00159         iy = v / v_step ;
00160 
00161         for ( h = 0 ; h < h_max ; h++ )
00162         {
00163             ix = N_BLK_HOR * iy + (h / h_step);
00164             *pImage++ = (gColorPattern[ix] & 0x0000FF);
00165             *pImage++ = (gColorPattern[ix] & 0x00FF00) >> 8;
00166             *pImage++ = (gColorPattern[ix] & 0xFF0000) >> 16;
00167         }
00168     }
00169 }
