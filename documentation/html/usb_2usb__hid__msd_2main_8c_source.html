00409 /**
00410  * Monitor keyboard buttons & Update key status in HID driver
00411  */
00412 static void HIDDKeyboardProcessKeys(void)
00413 {
00414     uint32_t i;
00415     uint8_t pressedKeys[NUM_KEYS];
00416     uint8_t pressedKeysSize = 0;
00417     uint8_t releasedKeys[NUM_KEYS];
00418     uint8_t releasedKeysSize = 0;
00419 
00420     /* Monitor buttons */
00421     for (i=0; i < PIO_LISTSIZE(pinsPushButtons); i++) {
00422 
00423         /* Check if button state has changed */
00424         uint8_t isButtonPressed = PIO_Get(&(pinsPushButtons[i]));
00425         if (isButtonPressed != keyStatus[i]) {
00426 
00427             /* Update button state */
00428             if (!isButtonPressed) {
00429 
00430                 /* Key has been pressed */
00431                 TRACE_INFO("-I- Key %u has been pressed\n\r", (unsigned int)i);
00432                 keyStatus[i] = 0;
00433                 pressedKeys[pressedKeysSize] = keyCodes[i];
00434                 pressedKeysSize++;
00435                 HIDDKeyboard_RemoteWakeUp();
00436             }
00437             else {
00438 
00439                 /* Key has been released */
00440                 TRACE_INFO("-I- Key %u has been released\n\r", (unsigned int)i);
00441                 keyStatus[i] = 1;
00442                 releasedKeys[releasedKeysSize] = keyCodes[i];
00443                 releasedKeysSize++;
00444             }
00445         }
00446     }
00447 
00448     /* Update key status in the HID driver if necessary */
00449     if ((pressedKeysSize != 0) || (releasedKeysSize != 0)) {
00450 
00451         uint8_t status;
00452 
00453         do {
00454 
00455             status = HIDDKeyboard_ChangeKeys(pressedKeys,
00456                                              pressedKeysSize,
00457                                              releasedKeys,
00458                                              releasedKeysSize);
00459         }
00460         while (status != USBD_STATUS_SUCCESS);
00461     }
00462 }
