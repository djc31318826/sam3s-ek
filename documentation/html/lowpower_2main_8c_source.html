00030 /**
00031  * \page lowpower Lowpower Example
00032  *
00033  * \section Purpose
00034  * This example allows to measure the consumption of the core in different modes
00035  * (active mode, sleep mode, wait mode, backup mode) of SAM3S.
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with SAM3S evaluation kits.
00040  *
00041  * \section Description
00042  *
00043  * At startup, the program configures all the PIOs as input to avoid
00044  * parasite consumption (except the UART PIOs). Then a menu is displayed.
00045  * It is divided into 2 parts. First part allows user to change the configuration
00046  * (MCK/PCK frequency configuration),
00047  * The second part allows user to enter in a mode To measure consumption.
00048  * An amperemeter has to be plugged on the board instead of the VIN jumper.
00049   *
00050  * \section Usage
00051  *
00052  * -# Build the program and download it inside the evaluation board. Please
00053  *    refer to the
00054  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6224.pdf">
00055  *    SAM-BA User Guide</a>, the
00056  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00057  *    GNU-Based Software Development</a> application note or to the
00058  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00059  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00060  * -# On the computer, open and configure a terminal application
00061  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00062  *   - 115200 bauds
00063  *   - 8 bits of data
00064  *   - No parity
00065  *   - 1 stop bit
00066  *   - No flow control
00067  * -# Start the application.
00068  * -# In the terminal window, the following text should appear:
00069  *    \code
00070  *     -- Lowpower Example xxx --
00071  *     -- xxxxxx-xx
00072  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00073  *    \endcode
00074  * -# Press one of the keys listed in the menu to perform the corresponding action.

00076  * \section References

00254 /**
00255  * \brief Restore working clock.
00256  * Here working clock must be running from PLL
00257  * and external crystal oscillator is used.
00258  */
00259 static void RestoreWorkingClock( uint32_t PllarVal, uint32_t PllbrVal, uint32_t MckrVal )
00260 {
00261     /* switch to slow clock first */
00262     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_SLOW_CLK ;
00263     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00264 
00265     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk) | PMC_MCKR_PRES_CLK ;
00266 
00267     /* Restart Main Oscillator */
00268     PMC->CKGR_MOR = (0x37 << 16) | (0x3F<<8) | CKGR_MOR_MOSCRCEN | CKGR_MOR_MOSCXTEN ;
00269     while ( !(PMC->PMC_SR & PMC_SR_MOSCXTS) ) ;
00270     /* Switch to moscsel */
00271     PMC->CKGR_MOR = (0x37 << 16) | (0x3F<<8) | CKGR_MOR_MOSCRCEN | CKGR_MOR_MOSCXTEN | CKGR_MOR_MOSCSEL ;
00272     while ( !(PMC->PMC_SR & PMC_SR_MOSCSELS) ) ;
00273 
00274     /* Switch to main oscillator */
00275     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK ;
00276     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00277 
00278     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk) | PMC_MCKR_PRES_CLK ;
00279     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00280 
00281     /* Restart PLL A */
00282     if ( (PllarVal & CKGR_PLLAR_MULA_Msk) != 0 )
00283     {
00284         PMC->CKGR_PLLAR = PllarVal ;
00285         while ( !(PMC->PMC_SR & PMC_SR_LOCKA) ) ;
00286     }
00287 
00288     /* Restart PLL B */
00289     if ( (PllbrVal & CKGR_PLLBR_MULB_Msk) != 0 )
00290     {
00291         PMC->CKGR_PLLBR = PllbrVal ;
00292         while ( !(PMC->PMC_SR & PMC_SR_LOCKB) ) ;
00293     }
00294 
00295     /* Switch to fast clock */
00296     PMC->PMC_MCKR = (MckrVal & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK ;
00297     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00298 
00299     PMC->PMC_MCKR = MckrVal ;
00300     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;

00303 /**
00304  * \brief Switch MCK to 32KHz RC Oscillator.
00305  *
00306  * \param pres Programmable Clock Prescaler
00307  */
00308 static void _SwitchMckTo32KRcOsc( uint32_t dwPres )
00309 {
00310     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_SLOW_CLK ;
00311     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00312 
00313     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk) | dwPres ;
00314     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00315 
00316     /* Stop PLL A */
00317     /* STMODE must be set at 2 when the PLLA is Off */
00318     PMC->CKGR_PLLAR = 0x2 << 14 ;
00319 
00320     /* Stop PLLB */
00321     /* STMODE must be set at 2 when the PLLB is OFF */
00322     PMC->CKGR_PLLBR = 0x2 << 14 ;
00323 
00324     /* Stop Main Oscillator */
00325     PMC->CKGR_MOR = (0x37 << 16) | (0x3F << 8) ;
00326 }

00328 /**
00329  * \brief Switch MCK to FastRC (Main On-Chip RC Oscillator).
00330  *
00331  * \param moscrcf Main On-Chip RC Oscillator Frequency Selection.
00332  * \param pres    Processor Clock Prescaler.
00333  */
00334 static void _SwitchMck2FastRC( uint32_t dwMoscrcf, uint32_t dwPres )
00335 {
00336     /* Enable Fast RC oscillator but DO NOT switch to RC now . Keep MOSCSEL to 1 */
00337     PMC->CKGR_MOR = CKGR_MOR_MOSCSEL | (0x37 << 16) | CKGR_MOR_MOSCXTEN | CKGR_MOR_MOSCRCEN ;
00338     /* Wait the Fast RC to stabilize */
00339     while ( !(PMC->PMC_SR & PMC_SR_MOSCRCS) ) ;
00340 
00341     /* Switch from Main Xtal osc to Fast RC */
00342     PMC->CKGR_MOR = (0x37 << 16) | CKGR_MOR_MOSCRCEN | CKGR_MOR_MOSCXTEN ;
00343     /* Wait for Main Oscillator Selection Status bit MOSCSELS */
00344     while ( !(PMC->PMC_SR & PMC_SR_MOSCSELS) ) ;
00345 
00346     /* Disable Main XTAL Oscillator */
00347     PMC->CKGR_MOR = (0x37 << 16) | CKGR_MOR_MOSCRCEN ;
00348 
00349     /* Change frequency of Fast RC oscillator */
00350     PMC->CKGR_MOR = (0x37 << 16) | PMC->CKGR_MOR | dwMoscrcf ;
00351     /* Wait the Fast RC to stabilize */
00352     while ( !(PMC->PMC_SR & PMC_SR_MOSCRCS) ) ;
00353 
00354     /* Switch to main clock */
00355     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK ;
00356     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00357     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_PRES_Msk) | dwPres ;
00358     while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) ) ;
00359 
00360     /* Stop PLL A */
00361     /* STMODE must be set at 2 when the PLLA is Off */
00362     PMC->CKGR_PLLAR = 0x2 << 14 ;
00363 
00364     /* Stop PLLB */
00365     /* STMODE must be set at 2 when the PLLB is OFF */
00366     PMC->CKGR_PLLBR = 0x2 << 14 ;
00367 }
