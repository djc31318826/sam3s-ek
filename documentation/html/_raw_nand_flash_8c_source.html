00121 /**
00122  * \brief Sends the column address to the NandFlash chip.
00123  *
00124  * \param raw  Pointer to a RawNandFlash instance.
00125  * \param columnAddress  Column address to send.
00126  */
00127 static void WriteColumnAddress(
00128     const struct RawNandFlash *raw,
00129     unsigned short columnAddress)
00130 {
00131     unsigned short pageDataSize = NandFlashModel_GetPageDataSize(MODEL(raw));
00132 
00133     /* Check the data bus width of the NandFlash */
00134     if (NandFlashModel_GetDataBusWidth(MODEL(raw)) == 16) {
00135         /* Div 2 is because we address in word and not in byte */
00136         columnAddress >>= 1;
00137     }
00138     /* Send single column address byte for small block devices, or two column address bytes for large block devices*/
00139     while (pageDataSize > 2) {
00140 
00141         if (NandFlashModel_GetDataBusWidth(MODEL(raw)) == 16) {
00142             WRITE_ADDRESS16(raw, columnAddress & 0xFF);
00143         }
00144         else {
00145             WRITE_ADDRESS(raw, columnAddress & 0xFF);
00146         }
00147         pageDataSize >>= 8;
00148         columnAddress >>= 8;
00149     }
00150 }

00320 /**
00321  * \brief Writes the data and/or the spare area of a page on a NandFlash chip.
00322  *
00323  * \param raw  Pointer to a RawNandFlash instance.
00324  * \param block  Number of the physical block to erase.
00325  * \param page  Number of the page to write inside the given block.
00326  * \param data  Buffer containing the data area.
00327  * \param spare  Buffer containing the spare area.
00328  * \return 0 if successful; otherwise returns an error code.
00329  * \note If one of the buffer pointer is 0, the corresponding area is not written.
00330  */
00331 static unsigned char WritePage(
00332     const struct RawNandFlash *raw,
00333     unsigned short block,
00334     unsigned short page,
00335     void *data,
00336     void *spare)
00337 {
00338     unsigned char error = 0;
00339     unsigned int pageDataSize = NandFlashModel_GetPageDataSize(MODEL(raw));
00340     unsigned int spareDataSize = NandFlashModel_GetPageSpareSize(MODEL(raw));
00341     unsigned short dummyByte;
00342     unsigned int rowAddress;
00343 
00344     TRACE_DEBUG("WritePage(B#%d:P#%d)\r\n", block, page);
00345     /* Calculate physical address of the page*/
00346     rowAddress = block * NandFlashModel_GetBlockSizeInPages(MODEL(raw)) + page;
00347 
00348     /* Start write operation*/
00349     ENABLE_CE(raw);
00350 
00351     /* Write data area if needed*/
00352     if (data) {
00353 
00354         WRITE_COMMAND(raw, COMMAND_WRITE_1);
00355         WriteColumnAddress(raw, 0);
00356         WriteRowAddress(raw, rowAddress);
00357         WriteData(raw, (unsigned char *) data, pageDataSize);
00358 
00359         /* Spare is written here as well since it is more efficient*/
00360         if (spare) {
00361 
00362             WriteData(raw, (unsigned char *) spare, spareDataSize);
00363         }
00364         else {
00365             /* Note: special case when ECC parity generation.
00366                ECC results are available as soon as the counter reaches the end of the main area.
00367                But when reach PageSize for an example, it could not generate last ECC_PR, The
00368                workaround is to receive PageSize+1 word.*/
00369             ReadData(raw, (unsigned char *) (&dummyByte), 2);
00370         }
00371         WRITE_COMMAND(raw, COMMAND_WRITE_2);
00372 
00373         WaitReady(raw);
00374         if (!IsOperationComplete(raw)) {
00375             TRACE_ERROR("WritePage: Failed writing data area.\n\r");
00376             error = NandCommon_ERROR_CANNOTWRITE;
00377         }
00378     }
00379 
00380     /* Write spare area alone if needed*/
00381     if (spare && !data) {
00382 
00383         WRITE_COMMAND(raw, COMMAND_WRITE_1);
00384         WriteColumnAddress(raw, pageDataSize);
00385         WriteRowAddress(raw, rowAddress);
00386         WriteData(raw, (unsigned char *) spare, spareDataSize);
00387         WRITE_COMMAND(raw, COMMAND_WRITE_2);
00388 
00389         WaitReady(raw);
00390         if (!IsOperationComplete(raw)) {
00391             TRACE_ERROR("WritePage: Failed writing data area.\n\r");
00392             error = NandCommon_ERROR_CANNOTWRITE;
00393         }
00394     }
00395 
00396     /* Disable chip*/
00397     DISABLE_CE(raw);
00398 
00399     return error;
00400 }

00403  * \brief Copies the data in a page of the NandFlash device to an other page on that same chip.
00404  *
00405  * \param raw  Pointer to a RawNandFlash instance.
00406  * \param sourceBlock  Source block number.
00407  * \param sourcePage  Source page number inside the source block.
00408  * \param destBlock  Destination block number.
00409  * \param destPage  Destination page number inside the destination block.
00410  * \return 0 if successful; otherwise returns an error code.
00411  * \note Both pages must have be even or odd; it is not possible to copy
00412  * and even page to an odd page and vice-versa.
00413  */
00414 static unsigned char CopyPage(
00415     const struct RawNandFlash *raw,
00416     unsigned short sourceBlock,
00417     unsigned short sourcePage,
00418     unsigned short destBlock,
00419     unsigned short destPage)
00420 {
00421     unsigned short numPages = NandFlashModel_GetBlockSizeInPages(MODEL(raw));
00422     unsigned int sourceRow = sourceBlock * numPages + sourcePage;
00423     unsigned int destRow = destBlock * numPages + destPage;
00424     unsigned char error = 0;
00425 
00426     assert((sourcePage & 1) == (destPage & 1) );
00427 
00428     TRACE_DEBUG("CopyPage(B#%d:P#%d -> B#%d:P#%d)\n\r",
00429               sourceBlock, sourcePage, destBlock, destPage);
00430 
00431     /* Use the copy-back facility if available*/
00432     if (NandFlashModel_SupportsCopyBack(MODEL(raw))) {
00433 
00434         /* Start operation*/
00435         ENABLE_CE(raw);
00436 
00437         /* Start copy-back read*/
00438         WRITE_COMMAND(raw, COMMAND_COPYBACK_READ_1);
00439         WriteColumnAddress(raw, 0);
00440         WriteRowAddress(raw, sourceRow);
00441         WRITE_COMMAND(raw, COMMAND_COPYBACK_READ_2);
00442         WaitReady(raw);
00443 
00444         /* Start copy-back write*/
00445         WRITE_COMMAND(raw, COMMAND_COPYBACK_PROGRAM_1);
00446         WriteColumnAddress(raw, 0);
00447         WriteRowAddress(raw, destRow);
00448         WRITE_COMMAND(raw, COMMAND_COPYBACK_PROGRAM_2);
00449         WaitReady(raw);
00450 
00451         /* Check status*/
00452         if (!IsOperationComplete(raw)) {
00453             TRACE_ERROR("CopyPage: Failed to copy page.\n\r");
00454             error = NandCommon_ERROR_CANNOTCOPY;
00455         }
00456 
00457         /* Finish operation*/
00458         DISABLE_CE(raw);
00459     }
00460     else {
00461 
00462         /* Software copy*/
00463         unsigned char data[NandCommon_MAXPAGEDATASIZE];

00466 
00467             TRACE_ERROR("CopyPage: Failed to read page to copy\n\r");

00471 
00472             TRACE_ERROR("CopyPage: Failed to write dest. page\n\r");
00473             error = NandCommon_ERROR_CANNOTWRITE;
