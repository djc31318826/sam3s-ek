00092     if ((maskedSR & STATUS_ERRORS) != 0)
00093     {
00094         pCommand->state = SDMMC_CMD_ERROR;
00095         /* Check error code */
00096         if ((maskedSR & STATUS_ERRORS) == HSMCI_SR_RTOE)
00097         {
00098             pCommand->status = SDMMC_ERROR_NORESPONSE;
00099         }
00100         else
00101         {

00111     {
00112         /* Command OK, disable interrupt */
00113         pMciHw->HSMCI_IDR = HSMCI_IDR_CMDRDY;
00114

00122             {
00123                 mask &= ~(uint32_t)HSMCI_IMR_CMDRDY;
00124             }

00126             {
00127                 pMciHw->HSMCI_IER = HSMCI_SR_NOTBUSY;
00128             }
00129         }
00130         else
00131         {
00132             mask &= ~(uint32_t)HSMCI_IMR_CMDRDY;
00133         }
00134     }
00135 
00136     /* Check if transfer stopped */
00137     if (maskedSR & HSMCI_SR_XFRDONE) {
00138         mask &= ~(uint32_t)HSMCI_IMR_XFRDONE;
00139     }
00140 
00141     /* Check if not busy */
00142     if (maskedSR & HSMCI_SR_NOTBUSY)
00143     {
00144         mask &= ~(uint32_t)HSMCI_IMR_NOTBUSY;

00149     {
00150         mask &= ~(uint32_t)HSMCI_IMR_RXBUFF;
00151     }
00152

00155     {
00156         mask &= ~(uint32_t)HSMCI_IMR_TXRDY;
00157     }

00181     }

00187         {
00188             mask &= ~(uint32_t)HSMCI_SR_BLKE;

00191         {
00192             pMciHw->HSMCI_IDR = HSMCI_SR_BLKE;
00193             pMciHw->HSMCI_IER = HSMCI_SR_TXRDY;
00194         }
00195     }
00196 
00197     /* Check if data write finished */
00198     if (maskedSR & HSMCI_SR_TXBUFE)
00199     {
00200         /* Data OK, disable interrupt */
00201         pMciHw->HSMCI_IDR = HSMCI_IDR_TXBUFE;

00203         /* End command until no data on bus */

00205         {
00206             pMciHw->HSMCI_IER = HSMCI_IER_FIFOEMPTY;
00207         }
00208         else
00209         {
00210             mask &= ~(uint32_t)HSMCI_SR_TXBUFE;
00211         }
00212     }
00213 
00214     /* All non-error mask done, complete the command */
00215     if (0 == mask || pCommand->state != SDMMC_CMD_PENDING)
00216     {
00217         /* Store the card response in the provided buffer */

00219         {
00220             switch (pCommand->resType)
00221             {
00222                 case 1:
00223                 case 3:
00224                 case 4:

00227                 case 7:

00230 
00231                 case 2:
00232                     resSize = 4;
00233                 break;
00234 
00235                 default:
00236                     resSize = 0;
00237                 break;
00238             }
00239 
00240             for (i=0; i < resSize; i++)
00241             {
00242                 pCommand->pResp[i] = pMciHw->HSMCI_RSPR[0];
00243             }
00244         }
00245 
00246         /* If no error occured, the transfer is successful */
00247         if (pCommand->state == SDMMC_CMD_PENDING)
00248         {
00249             pCommand->state = 0;
00250         }
00251         /* Error ? Reset to remove transfer statuses */
00252         else
00253         {
