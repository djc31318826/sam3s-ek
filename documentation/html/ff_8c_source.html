00489 DWORD get_fat ( /* 0xFFFFFFFF:Disk error, 1:Internal error, Else:Cluster status */
00490     FATFS *fs,  /* File system object */
00491     DWORD clst  /* Cluster# to get the link information */
00492 )
00493 {
00494     UINT wc, bc;
00495     BYTE *p;
00496 
00497 
00498     if (clst < 2 || clst >= fs->n_fatent)   /* Chack range */
00499         return 1;
00500 
00501     switch (fs->fs_type) {
00502     case FS_FAT12 :
00503         bc = (UINT)clst; bc += bc / 2;
00504         if (move_window(fs, fs->fatbase + (bc / SS(fs)))) break;
00505         wc = fs->win[bc % SS(fs)]; bc++;
00506         if (move_window(fs, fs->fatbase + (bc / SS(fs)))) break;
00507         wc |= fs->win[bc % SS(fs)] << 8;
00508         return (clst & 1) ? (wc >> 4) : (wc & 0xFFF);
00509 
00510     case FS_FAT16 :
00511         if (move_window(fs, fs->fatbase + (clst / (SS(fs) / 2)))) break;
00512         p = &fs->win[clst * 2 % SS(fs)];
00513         return LD_WORD(p);
00514 
00515     case FS_FAT32 :
00516         if (move_window(fs, fs->fatbase + (clst / (SS(fs) / 4)))) break;
00517         p = &fs->win[clst * 4 % SS(fs)];
00518         return LD_DWORD(p) & 0x0FFFFFFF;
00519     }
00520 
00521     return 0xFFFFFFFF;  /* An error occurred at the disk I/O layer */
00522 }

01267 static
01268 FRESULT create_name (
01269     DIR *dj,            /* Pointer to the directory object */
01270     const TCHAR **path  /* Pointer to pointer to the segment in the path string */
01271 )
01272 {
01273 #ifdef _EXCVT
01274     static const BYTE excvt[] = _EXCVT; /* Upper conversion table for extended chars */
01275 #endif
01276 
01277 #if _USE_LFN    /* LFN configuration */
01278     BYTE b, cf;
01279     WCHAR w, *lfn;
01280     int i, ni, si, di;
01281     const TCHAR *p;
01282 
01283     /* Create LFN in Unicode */
01284     si = di = 0;
01285     p = *path;
01286     lfn = dj->lfn;
01287     for (;;) {
01288         w = p[si++];                    /* Get a character */
01289         if (w < ' ' || w == '/' || w == '\\') break;    /* Break on end of segment */
01290         if (di >= _MAX_LFN)             /* Reject too long name */
01291             return FR_INVALID_NAME;
01292 #if !_LFN_UNICODE
01293         w &= 0xFF;
01294         if (IsDBCS1(w)) {               /* If it is a DBC 1st byte */
01295             b = p[si++];                /* Get 2nd byte */
01296             if (!IsDBCS2(b))            /* Reject invalid code for DBC */
01297                 return FR_INVALID_NAME;
01298             w = (w << 8) + b;
01299         }
01300         w = ff_convert(w, 1);           /* Convert OEM to Unicode */
01301         if (!w) return FR_INVALID_NAME; /* Reject invalid code */
01302 #endif
01303         if (w < 0x80 && chk_chr("\"*:<>\?|\x7F", w)) /* Reject illegal chars for LFN */
01304             return FR_INVALID_NAME;
01305         lfn[di++] = w;                  /* Store the Unicode char */
01306     }
01307     *path = &p[si];                     /* Return pointer to the next segment */
01308     cf = (w < ' ') ? NS_LAST : 0;       /* Set last segment flag if end of path */
01309 #if _FS_RPATH
01310     if ((di == 1 && lfn[di - 1] == '.') || /* Is this a dot entry? */
01311         (di == 2 && lfn[di - 1] == '.' && lfn[di - 2] == '.')) {
01312         lfn[di] = 0;
01313         for (i = 0; i < 11; i++)
01314             dj->fn[i] = (i < di) ? '.' : ' ';
01315         dj->fn[i] = cf | NS_DOT;        /* This is a dot entry */
01316         return FR_OK;
01317     }
01318 #endif
01319     while (di) {                        /* Strip trailing spaces and dots */
01320         w = lfn[di - 1];
01321         if (w != ' ' && w != '.') break;
01322         di--;
01323     }
01324     if (!di) return FR_INVALID_NAME;    /* Reject nul string */
01325 
01326     lfn[di] = 0;                        /* LFN is created */
01327 
01328     /* Create SFN in directory form */
01329     mem_set(dj->fn, ' ', 11);
01330     for (si = 0; lfn[si] == ' ' || lfn[si] == '.'; si++) ;  /* Strip leading spaces and dots */
01331     if (si) cf |= NS_LOSS | NS_LFN;
01332     while (di && lfn[di - 1] != '.') di--;  /* Find extension (di<=si: no extension) */
01333 
01334     b = i = 0; ni = 8;
01335     for (;;) {
01336         w = lfn[si++];                  /* Get an LFN char */
01337         if (!w) break;                  /* Break on end of the LFN */
01338         if (w == ' ' || (w == '.' && si != di)) {   /* Remove spaces and dots */
01339             cf |= NS_LOSS | NS_LFN; continue;
01340         }
01341 
01342         if (i >= ni || si == di) {      /* Extension or end of SFN */
01343             if (ni == 11) {             /* Long extension */
01344                 cf |= NS_LOSS | NS_LFN; break;
01345             }
01346             if (si != di) cf |= NS_LOSS | NS_LFN;   /* Out of 8.3 format */
01347             if (si > di) break;         /* No extension */
01348             si = di; i = 8; ni = 11;    /* Enter extension section */
01349             b <<= 2; continue;
01350         }
01351 
01352         if (w >= 0x80) {                /* Non ASCII char */
01353 #ifdef _EXCVT
01354             w = ff_convert(w, 0);       /* Unicode -> OEM code */
01355             if (w) w = excvt[w - 0x80]; /* Convert extended char to upper (SBCS) */
01356 #else
01357             w = ff_convert(ff_wtoupper(w), 0);  /* Upper converted Unicode -> OEM code */
01358 #endif
01359             cf |= NS_LFN;               /* Force create LFN entry */
01360         }
01361 
01362         if (_DF1S && w >= 0x100) {      /* Double byte char */
01363             if (i >= ni - 1) {
01364                 cf |= NS_LOSS | NS_LFN; i = ni; continue;
01365             }
01366             dj->fn[i++] = (BYTE)(w >> 8);
01367         } else {                        /* Single byte char */
01368             if (!w || chk_chr("+,;=[]", w)) {       /* Replace illegal chars for SFN */
01369                 w = '_'; cf |= NS_LOSS | NS_LFN;    /* Lossy conversion */
01370             } else {
01371                 if (IsUpper(w)) {       /* ASCII large capital */
01372                     b |= 2;
01373                 } else {
01374                     if (IsLower(w)) {   /* ASCII small capital */
01375                         b |= 1; w -= 0x20;
01376                     }
01377                 }
01378             }
01379         }
01380         dj->fn[i++] = (BYTE)w;
01381     }
01382 
01383     if (dj->fn[0] == 0xE5) dj->fn[0] = 0x05;    /* If the first char collides with deleted mark, replace it with 0x05 */
01384 
01385     if (ni == 8) b <<= 2;
01386     if ((b & 0x0C) == 0x0C || (b & 0x03) == 0x03)   /* Create LFN entry when there are composite capitals */
01387         cf |= NS_LFN;
01388     if (!(cf & NS_LFN)) {                       /* When LFN is in 8.3 format without extended char, NT flags are created */
01389         if ((b & 0x03) == 0x01) cf |= NS_EXT;   /* NT flag (Extension has only small capital) */
01390         if ((b & 0x0C) == 0x04) cf |= NS_BODY;  /* NT flag (Filename has only small capital) */
01391     }
01392 
01393     dj->fn[NS] = cf;    /* SFN is created */
01394 
01395     return FR_OK;
01396 
01397 
01398 #else   /* Non-LFN configuration */
01399     BYTE b, c, d, *sfn;
01400     int ni, si, i;
01401     const char *p;
01402 
01403     /* Create file name in directory form */
01404     sfn = dj->fn;
01405     mem_set(sfn, ' ', 11);
01406     si = i = b = 0; ni = 8;
01407     p = *path;
01408 #if _FS_RPATH
01409     if (p[si] == '.') { /* Is this a dot entry? */
01410         for (;;) {
01411             c = (BYTE)p[si++];
01412             if (c != '.' || si >= 3) break;
01413             sfn[i++] = c;
01414         }
01415         if (c != '/' && c != '\\' && c > ' ') return FR_INVALID_NAME;
01416         *path = &p[si];                                 /* Return pointer to the next segment */
01417         sfn[NS] = (c <= ' ') ? NS_LAST | NS_DOT : NS_DOT;   /* Set last segment flag if end of path */
01418         return FR_OK;
01419     }
01420 #endif
01421     for (;;) {
01422         c = (BYTE)p[si++];
01423         if (c <= ' ' || c == '/' || c == '\\') break;   /* Break on end of segment */
01424         if (c == '.' || i >= ni) {
01425             if (ni != 8 || c != '.') return FR_INVALID_NAME;
01426             i = 8; ni = 11;
01427             b <<= 2; continue;
01428         }
01429         if (c >= 0x80) {                /* Extended char */
01430 #ifdef _EXCVT
01431             c = excvt[c - 0x80];        /* Convert extend char (SBCS) */
01432 #else
01433             b |= 3;                     /* Eliminate NT flag if extended char is exist */
01434 #if !_DF1S  /* ASCII only cfg */
01435             return FR_INVALID_NAME;
01436 #endif
01437 #endif
01438         }
01439         if (IsDBCS1(c)) {               /* DBC 1st byte? */
01440             d = (BYTE)p[si++];          /* Get 2nd byte */
01441             if (!IsDBCS2(d) || i >= ni - 1) /* Reject invalid DBC */
01442                 return FR_INVALID_NAME;
01443             sfn[i++] = c;
01444             sfn[i++] = d;
01445         } else {                        /* Single byte code */
01446             if (chk_chr("\"*+,:<=>\?[]|\x7F", c))   /* Reject illegal chrs for SFN */
01447                 return FR_INVALID_NAME;
01448             if (IsUpper(c)) {           /* ASCII large capital? */
01449                 b |= 2;
01450             } else {
01451                 if (IsLower(c)) {       /* ASCII small capital? */
01452                     b |= 1; c -= 0x20;
01453                 }
01454             }
01455             sfn[i++] = c;
01456         }
01457     }
01458     *path = &p[si];                     /* Return pointer to the next segment */
01459     c = (c <= ' ') ? NS_LAST : 0;       /* Set last segment flag if end of path */
01460 
01461     if (!i) return FR_INVALID_NAME;     /* Reject nul string */
01462     if (sfn[0] == 0xE5) sfn[0] = 0x05;  /* When first char collides with 0xE5, replace it with 0x05 */
01463 
01464     if (ni == 8) b <<= 2;
01465     if ((b & 0x03) == 0x01) c |= NS_EXT;    /* NT flag (Name extension has only small capital) */
01466     if ((b & 0x0C) == 0x04) c |= NS_BODY;   /* NT flag (Name body has only small capital) */
01467 
01468     sfn[NS] = c;        /* Store NT flag, File name is created */
01469 
01470     return FR_OK;
01471 #endif
01472 }

01650 /*-----------------------------------------------------------------------*/
01651 
01652 static
01653 FRESULT chk_mounted (   /* FR_OK(0): successful, !=0: any error occurred */
01654     const TCHAR **path, /* Pointer to pointer to the path name (drive number) */
01655     FATFS **rfs,        /* Pointer to pointer to the found file system object */
01656     BYTE chk_wp         /* !=0: Check media write protection for write access */
01657 )
01658 {
01659     BYTE fmt, b, *tbl;
01660     UINT vol;
01661     DSTATUS stat;
01662     DWORD bsect, fasize, tsect, sysect, nclst, szbfat;
01663     WORD nrsv;
01664     const TCHAR *p = *path;
01665     FATFS *fs;
01666 
01667     /* Get logical drive number from the path name */
01668     vol = p[0] - '0';               /* Is there a drive number? */
01669     if (vol <= 9 && p[1] == ':') {  /* Found a drive number, get and strip it */
01670         p += 2; *path = p;          /* Return pointer to the path name */
01671     } else {                        /* No drive number is given */
01672 #if _FS_RPATH
01673         vol = Drive;                /* Use current drive */
01674 #else
01675         vol = 0;                    /* Use drive 0 */
01676 #endif
01677     }
01678 
01679     /* Check if the logical drive is valid or not */
01680     if (vol >= _DRIVES)             /* Is the drive number valid? */
01681         return FR_INVALID_DRIVE;
01682     *rfs = fs = FatFs[vol];         /* Return pointer to the corresponding file system object */
01683     if (!fs) return FR_NOT_ENABLED; /* Is the file system object available? */
01684 
01685     ENTER_FF(fs);                   /* Lock file system */
01686 
01687     if (fs->fs_type) {              /* If the logical drive has been mounted */
01688         stat = disk_status(fs->drv);
01689         if (!(stat & STA_NOINIT)) { /* and the physical drive is kept initialized (has not been changed), */
01690 #if !_FS_READONLY
01691             if (chk_wp && (stat & STA_PROTECT)) /* Check write protection if needed */
01692                 return FR_WRITE_PROTECTED;
01693 #endif
01694             return FR_OK;           /* The file system object is valid */
01695         }
01696     }
01697 
01698     /* The logical drive must be mounted. Following code attempts to mount the volume (initialize the file system object) */
01699 
01700     fs->fs_type = 0;                    /* Clear the file system object */
01701     fs->drv = (BYTE)LD2PD(vol);         /* Bind the logical drive and a physical drive */
01702     stat = disk_initialize(fs->drv);    /* Initialize low level disk I/O layer */
01703     if (stat & STA_NOINIT)              /* Check if the drive is ready */
01704         return FR_NOT_READY;
01705 #if _MAX_SS != 512                      /* Get disk sector size if needed */
01706     if (disk_ioctl(fs->drv, GET_SECTOR_SIZE, &SS(fs)) != RES_OK || SS(fs) > _MAX_SS)
01707         return FR_NO_FILESYSTEM;
01708 #endif
01709 #if !_FS_READONLY
01710     if (chk_wp && (stat & STA_PROTECT)) /* Check disk write protection if needed */
01711         return FR_WRITE_PROTECTED;
01712 #endif
01713     /* Search FAT partition on the drive (Supports only generic partitionings, FDISK and SFD) */
01714     fmt = check_fs(fs, bsect = 0);      /* Check sector 0 if it is a VBR */
01715     if (fmt == 1) {                     /* Not an FAT-VBR, the disk may be partitioned */
01716         /* Check the partition listed in top of the partition table */
01717         tbl = &fs->win[MBR_Table + LD2PT(vol) * 16];    /* Partition table */
01718         if (tbl[4]) {                                   /* Is the partition existing? */
01719             bsect = LD_DWORD(&tbl[8]);                  /* Partition offset in LBA */
01720             fmt = check_fs(fs, bsect);                  /* Check the partition */
01721         }
01722     }
01723     if (fmt == 3) return FR_DISK_ERR;
01724     if (fmt) return FR_NO_FILESYSTEM;                   /* No FAT volume is found */
01725 
01726     /* Following code initializes the file system object */
01727 
01728     if (LD_WORD(fs->win+BPB_BytsPerSec) != SS(fs))      /* (BPB_BytsPerSec must be equal to the physical sector size) */
01729         return FR_NO_FILESYSTEM;
01730 
01731     fasize = LD_WORD(fs->win+BPB_FATSz16);              /* Number of sectors per FAT */
01732     if (!fasize) fasize = LD_DWORD(fs->win+BPB_FATSz32);
01733     fs->fsize = fasize;
01734 
01735     fs->n_fats = b = fs->win[BPB_NumFATs];              /* Number of FAT copies */
01736     if (b != 1 && b != 2) return FR_NO_FILESYSTEM;      /* (Must be 1 or 2) */
01737     fasize *= b;                                        /* Number of sectors for FAT area */
01738 
01739     fs->csize = b = fs->win[BPB_SecPerClus];            /* Number of sectors per cluster */
01740     if (!b || (b & (b - 1))) return FR_NO_FILESYSTEM;   /* (Must be 1,2,4...128) */
01741 
01742     fs->n_rootdir = LD_WORD(fs->win+BPB_RootEntCnt);    /* Number of root directory entries */
01743     if (fs->n_rootdir % (SS(fs) / 32)) return FR_NO_FILESYSTEM; /* (BPB_RootEntCnt must be sector aligned) */
01744 
01745     tsect = LD_WORD(fs->win+BPB_TotSec16);              /* Number of sectors on the volume */
01746     if (!tsect) tsect = LD_DWORD(fs->win+BPB_TotSec32);
01747 
01748     nrsv = LD_WORD(fs->win+BPB_RsvdSecCnt);             /* Number of reserved sectors */
01749     if (!nrsv) return FR_NO_FILESYSTEM;                 /* (BPB_RsvdSecCnt must not be 0) */
01750 
01751     /* Determine the FAT sub type */

01774     if (fs->fsize < (szbfat + (SS(fs) - 1)) / SS(fs))   /* (FAT size must not be less than FAT sectors */

01777 #if !_FS_READONLY

01787             LD_WORD(fs->win+BS_55AA) == 0xAA55 &&
01788             LD_DWORD(fs->win+FSI_LeadSig) == 0x41615252 &&
01789             LD_DWORD(fs->win+FSI_StrucSig) == 0x61417272) {

01791                 fs->free_clust = LD_DWORD(fs->win+FSI_Free_Count);
01792         }
01793     }
01794 #endif

01796     fs->id = ++Fsid;        /* File system mount ID */
01797     fs->winsect = 0;        /* Invalidate sector cache */
01798     fs->wflag = 0;
01799 #if _FS_RPATH
01800     fs->cdir = 0;           /* Current directory (root dir) */
01801 #endif

02018 FRESULT f_read (
02019     FIL *fp,        /* Pointer to the file object */
02020     void *buff,     /* Pointer to data buffer */
02021     UINT btr,       /* Number of bytes to read */
02022     UINT *br        /* Pointer to number of bytes read */
02023 )
02024 {
02025     FRESULT res;
02026     DWORD clst, sect, remain;
02027     UINT rcnt, cc;
02028     BYTE csect, *rbuff = buff;
02029 
02030 
02031     *br = 0;    /* Initialize byte counter */
02032 
02033     res = validate(fp->fs, fp->id);                 /* Check validity of the object */
02034     if (res != FR_OK) LEAVE_FF(fp->fs, res);
02035     if (fp->flag & FA__ERROR)                       /* Check abort flag */
02036         LEAVE_FF(fp->fs, FR_INT_ERR);
02037     if (!(fp->flag & FA_READ))                      /* Check access mode */
02038         LEAVE_FF(fp->fs, FR_DENIED);
02039     remain = fp->fsize - fp->fptr;
02040     if (btr > remain) btr = (UINT)remain;           /* Truncate btr by remaining bytes */
02041 
02042     for ( ;  btr;                                   /* Repeat until all data transferred */
02043         rbuff += rcnt, fp->fptr += rcnt, *br += rcnt, btr -= rcnt) {
02044         if ((fp->fptr % SS(fp->fs)) == 0) {         /* On the sector boundary? */
02045             csect = (BYTE)(fp->fptr / SS(fp->fs) & (fp->fs->csize - 1));    /* Sector offset in the cluster */
02046             if (!csect) {                           /* On the cluster boundary? */
02047                 clst = (fp->fptr == 0) ?            /* On the top of the file? */
02048                     fp->org_clust : get_fat(fp->fs, fp->curr_clust);
02049                 if (clst <= 1) ABORT(fp->fs, FR_INT_ERR);
02050                 if (clst == 0xFFFFFFFF) ABORT(fp->fs, FR_DISK_ERR);
02051                 fp->curr_clust = clst;              /* Update current cluster */
02052             }
02053             sect = clust2sect(fp->fs, fp->curr_clust);  /* Get current sector */
02054             if (!sect) ABORT(fp->fs, FR_INT_ERR);
02055             sect += csect;
02056             cc = btr / SS(fp->fs);                  /* When remaining bytes >= sector size, */
02057             if (cc) {                               /* Read maximum contiguous sectors directly */
02058                 if (csect + cc > fp->fs->csize)     /* Clip at cluster boundary */
02059                     cc = fp->fs->csize - csect;
02060                 if (disk_read(fp->fs->drv, rbuff, sect, (BYTE)cc) != RES_OK)
02061                     ABORT(fp->fs, FR_DISK_ERR);
02062 #if !_FS_READONLY && _FS_MINIMIZE <= 2              /* Replace one of the read sectors with cached data if it contains a dirty sector */
02063 #if _FS_TINY
02064                 if (fp->fs->wflag && fp->fs->winsect - sect < cc)
02065                     mem_cpy(rbuff + ((fp->fs->winsect - sect) * SS(fp->fs)), fp->fs->win, SS(fp->fs));
02066 #else
02067                 if ((fp->flag & FA__DIRTY) && fp->dsect - sect < cc)
02068                     mem_cpy(rbuff + ((fp->dsect - sect) * SS(fp->fs)), fp->buf, SS(fp->fs));
02069 #endif
02070 #endif
02071                 rcnt = SS(fp->fs) * cc;             /* Number of bytes transferred */
02072                 continue;
02073             }
02074 #if !_FS_TINY
02075 #if !_FS_READONLY
02076             if (fp->flag & FA__DIRTY) {         /* Write sector I/O buffer if needed */
02077                 if (disk_write(fp->fs->drv, fp->buf, fp->dsect, 1) != RES_OK)
02078                     ABORT(fp->fs, FR_DISK_ERR);
02079                 fp->flag &= ~FA__DIRTY;
02080             }
02081 #endif
02082             if (fp->dsect != sect) {            /* Fill sector buffer with file data */
02083                 if (disk_read(fp->fs->drv, fp->buf, sect, 1) != RES_OK)
02084                     ABORT(fp->fs, FR_DISK_ERR);
02085             }
02086 #endif
02087             fp->dsect = sect;
02088         }
02089         rcnt = SS(fp->fs) - (fp->fptr % SS(fp->fs));    /* Get partial sector data from sector buffer */
02090         if (rcnt > btr) rcnt = btr;
02091 #if _FS_TINY
02092         if (move_window(fp->fs, fp->dsect))         /* Move sector window */
02093             ABORT(fp->fs, FR_DISK_ERR);
02094         mem_cpy(rbuff, &fp->fs->win[fp->fptr % SS(fp->fs)], rcnt);  /* Pick partial sector */
02095 #else
02096         mem_cpy(rbuff, &fp->buf[fp->fptr % SS(fp->fs)], rcnt);  /* Pick partial sector */
02097 #endif
02098     }
02099 
02100     LEAVE_FF(fp->fs, FR_OK);
02101 }

02369 FRESULT f_lseek (
02370     FIL *fp,        /* Pointer to the file object */
02371     DWORD ofs       /* File pointer from top of file */
02372 )

02375 
02376 
02377     res = validate(fp->fs, fp->id);     /* Check validity of the object */
02378     if (res != FR_OK) LEAVE_FF(fp->fs, res);
02379     if (fp->flag & FA__ERROR)           /* Check abort flag */
02380         LEAVE_FF(fp->fs, FR_INT_ERR);
02381 
02382 #if _USE_FASTSEEK
02383     if (fp->cltbl) {    /* Fast seek */
02384         DWORD cl, pcl, ncl, tcl, dsc, tlen, *tbl = fp->cltbl;
02385         BYTE csc;
02386 
02387         tlen = *tbl++;
02388         if (ofs == CREATE_LINKMAP) {    /* Create link map table */
02389             cl = fp->org_clust;
02390             if (cl) {
02391                 do {
02392                     if (tlen < 4) { /* Not enough table items */
02393                         res = FR_NOT_ENOUGH_CORE; break;
02394                     }
02395                     tcl = cl; ncl = 0;
02396                     do {        /* Get a fragment and store the top and length */
02397                         pcl = cl; ncl++;
02398                         cl = get_fat(fp->fs, cl);
02399                         if (cl <= 1) ABORT(fp->fs, FR_INT_ERR);
02400                         if (cl == 0xFFFFFFFF) ABORT(fp->fs, FR_DISK_ERR);
02401                     } while (cl == pcl + 1);
02402                     *tbl++ = ncl; *tbl++ = tcl;
02403                     tlen -= 2;
02404                 } while (cl < fp->fs->n_fatent);
02405             }
02406             *tbl = 0;   /* Terminate table */
02407 
02408         } else {                        /* Fast seek */
02409             if (ofs > fp->fsize)        /* Clip offset at the file size */
02410                 ofs = fp->fsize;
02411             fp->fptr = ofs;             /* Set file pointer */
02412             if (ofs) {
02413                 dsc = (ofs - 1) / SS(fp->fs);
02414                 cl = dsc / fp->fs->csize;
02415                 for (;;) {
02416                     ncl = *tbl++;
02417                     if (!ncl) ABORT(fp->fs, FR_INT_ERR);
02418                     if (cl < ncl) break;
02419                     cl -= ncl; tbl++;
02420                 }
02421                 fp->curr_clust = cl + *tbl;
02422                 csc = (BYTE)(dsc & (fp->fs->csize - 1));
02423                 dsc = clust2sect(fp->fs, fp->curr_clust);
02424                 if (!dsc) ABORT(fp->fs, FR_INT_ERR);
02425                 dsc += csc;
02426                 if (fp->fptr % SS(fp->fs) && dsc != fp->dsect) {
02427 #if !_FS_TINY
02428 #if !_FS_READONLY
02429                     if (fp->flag & FA__DIRTY) {     /* Flush dirty buffer if needed */
02430                         if (disk_write(fp->fs->drv, fp->buf, fp->dsect, 1) != RES_OK)
02431                             ABORT(fp->fs, FR_DISK_ERR);
02432                         fp->flag &= ~FA__DIRTY;
02433                     }
02434 #endif
02435                     if (disk_read(fp->fs->drv, fp->buf, dsc, 1) != RES_OK)
02436                         ABORT(fp->fs, FR_DISK_ERR);
02437 #endif
02438                     fp->dsect = dsc;
02439                 }
02440             }
02441         }
02442     } else
02443 #endif
02444 
02445     /* Normal Seek */
02446     {
02447         DWORD clst, bcs, nsect, ifptr;
02448 
02449         if (ofs > fp->fsize                 /* In read-only mode, clip offset with the file size */
02450 #if !_FS_READONLY
02451              && !(fp->flag & FA_WRITE)
02452 #endif
02453             ) ofs = fp->fsize;
02454 
02455         ifptr = fp->fptr;
02456         fp->fptr = nsect = 0;
02457         if (ofs) {
02458             bcs = (DWORD)fp->fs->csize * SS(fp->fs);    /* Cluster size (byte) */
02459             if (ifptr > 0 &&
02460                 (ofs - 1) / bcs >= (ifptr - 1) / bcs) { /* When seek to same or following cluster, */
02461                 fp->fptr = (ifptr - 1) & ~(bcs - 1);    /* start from the current cluster */
02462                 ofs -= fp->fptr;
02463                 clst = fp->curr_clust;
02464             } else {                                    /* When seek to back cluster, */
02465                 clst = fp->org_clust;                   /* start from the first cluster */
02466 #if !_FS_READONLY
02467                 if (clst == 0) {                        /* If no cluster chain, create a new chain */
02468                     clst = create_chain(fp->fs, 0);
02469                     if (clst == 1) ABORT(fp->fs, FR_INT_ERR);
02470                     if (clst == 0xFFFFFFFF) ABORT(fp->fs, FR_DISK_ERR);
02471                     fp->org_clust = clst;
02472                 }
02473 #endif
02474                 fp->curr_clust = clst;
02475             }
02476             if (clst != 0) {
02477                 while (ofs > bcs) {                     /* Cluster following loop */
02478 #if !_FS_READONLY
02479                     if (fp->flag & FA_WRITE) {          /* Check if in write mode or not */
02480                         clst = create_chain(fp->fs, clst);  /* Force stretch if in write mode */
02481                         if (clst == 0) {                /* When disk gets full, clip file size */
02482                             ofs = bcs; break;
02483                         }
02484                     } else
02485 #endif
02486                         clst = get_fat(fp->fs, clst);   /* Follow cluster chain if not in write mode */
02487                     if (clst == 0xFFFFFFFF) ABORT(fp->fs, FR_DISK_ERR);
02488                     if (clst <= 1 || clst >= fp->fs->n_fatent) ABORT(fp->fs, FR_INT_ERR);
02489                     fp->curr_clust = clst;
02490                     fp->fptr += bcs;
02491                     ofs -= bcs;
02492                 }
02493                 fp->fptr += ofs;
02494                 if (ofs % SS(fp->fs)) {
02495                     nsect = clust2sect(fp->fs, clst);   /* Current sector */
02496                     if (!nsect) ABORT(fp->fs, FR_INT_ERR);
02497                     nsect += ofs / SS(fp->fs);
02498                 }
02499             }
02500         }
02501         if (fp->fptr % SS(fp->fs) && nsect != fp->dsect) {
02502 #if !_FS_TINY
02503 #if !_FS_READONLY
02504             if (fp->flag & FA__DIRTY) {         /* Flush dirty buffer if needed */
02505                 if (disk_write(fp->fs->drv, fp->buf, fp->dsect, 1) != RES_OK)
02506                     ABORT(fp->fs, FR_DISK_ERR);
02507                 fp->flag &= ~FA__DIRTY;
02508             }
02509 #endif
02510             if (disk_read(fp->fs->drv, fp->buf, nsect, 1) != RES_OK)
02511                 ABORT(fp->fs, FR_DISK_ERR);
02512 #endif
02513             fp->dsect = nsect;
02514         }
02515 #if !_FS_READONLY
02516         if (fp->fptr > fp->fsize) {         /* Set changed flag if the file size is extended */
02517             fp->fsize = fp->fptr;
02518             fp->flag |= FA__WRITTEN;
02519         }
02520 #endif
02521     }
02522 
02523     LEAVE_FF(fp->fs, res);
02524 }

03133 FRESULT f_mkfs (
03134     BYTE drv,       /* Logical drive number */
03135     BYTE sfd,       /* Partitioning rule 0:FDISK, 1:SFD */
03136     UINT au         /* Allocation unit size [bytes] */
03137 )
03138 {
03139     static const WORD vst[] = { 1024,   512,  256,  128,   64,    32,   16,    8,    4,    2,   0};
03140     static const WORD cst[] = {32768, 16384, 8192, 4096, 2048, 16384, 8192, 4096, 2048, 1024, 512};
03141     BYTE fmt, md, *tbl;
03142     DWORD n_clst, vs, n;
03143     UINT as, i;
03144     DWORD b_vol, b_fat, b_dir, b_data;      /* Area offset (LBA) */
03145     DWORD n_vol, n_rsv, n_fat, n_dir;       /* Area size */
03146     FATFS *fs;
03147     DSTATUS stat;
03148 
03149 
03150     /* Check mounted drive and clear work area */
03151     if (drv >= _DRIVES) return FR_INVALID_DRIVE;
03152     fs = FatFs[drv];
03153     if (!fs) return FR_NOT_ENABLED;
03154     fs->fs_type = 0;
03155     drv = LD2PD(drv);
03156 
03157     /* Get disk statics */
03158     stat = disk_initialize(drv);
03159     if (stat & STA_NOINIT) return FR_NOT_READY;
03160     if (stat & STA_PROTECT) return FR_WRITE_PROTECTED;
03161 #if _MAX_SS != 512                  /* Get disk sector size */

03163         return FR_DISK_ERR;
03164 #endif
03165     if (disk_ioctl(drv, GET_SECTOR_COUNT, &n_vol) != RES_OK || n_vol < 128)
03166         return FR_DISK_ERR;

03168     n_vol -= b_vol;
03169     if (au & (au - 1)) au = 0;      /* Check validity of the allocation unit size */
03170     if (!au) {                      /* AU auto selection */
03171         vs = n_vol / (2000 / (SS(fs) / 512));
03172         for (i = 0; vs < vst[i]; i++) ;
03173         au = cst[i];
03174     }
03175     if (_MAX_SS != 512 && au < SS(fs)) au = SS(fs);
03176     au /= SS(fs);       /* Number of sectors per cluster */
03177     if (au == 0) au = 1;
03178     if (au > 128) au = 128;
03179 
03180     /* Pre-compute number of clusters and FAT syb-type */
03181     n_clst = n_vol / au;
03182     fmt = FS_FAT12;
03183     if (n_clst >= MIN_FAT16) fmt = FS_FAT16;
03184     if (n_clst >= MIN_FAT32) fmt = FS_FAT32;
03185 
03186     /* Determine offset and size of FAT structure */
03187     if (fmt == FS_FAT32) {
03188         n_fat = ((n_clst * 4) + 8 + SS(fs) - 1) / SS(fs);
03189         n_rsv = 32;
03190         n_dir = 0;
03191     } else {
03192         n_fat = (fmt == FS_FAT12) ? (n_clst * 3 + 1) / 2 + 3 : (n_clst * 2) + 4;
03193         n_fat = (n_fat + SS(fs) - 1) / SS(fs);
03194         n_rsv = 1;
03195         n_dir = N_ROOTDIR * 32UL / SS(fs);
03196     }
03197     b_fat = b_vol + n_rsv;              /* FAT area start sector */
03198     b_dir = b_fat + n_fat * N_FATS;     /* Directory area start sector */
03199     b_data = b_dir + n_dir;             /* Data area start sector */
03200     if (n_vol < b_data + au) return FR_MKFS_ABORTED;    /* Too small volume */
03201 
03202     /* Align data start sector to erase block boundary (for flash memory media) */
03203     if (disk_ioctl(drv, GET_BLOCK_SIZE, &n) != RES_OK) return FR_DISK_ERR;
03204     if (!n || n > 32768) return FR_MKFS_ABORTED;
03205     n = (b_data + n - 1) & ~(n - 1);    /* Next nearest boundary from current data start */
03206     n = (n - b_data) / N_FATS;
03207     if (fmt == FS_FAT32) {      /* FAT32: Move FAT start */
03208         n_rsv += n;
03209         b_fat += n;
03210     } else {                    /* FAT12/16: Expand FAT size */
03211         n_fat += n;
03212     }
03213     /* b_dir and b_data are no longer used below */
03214 
03215     /* Determine number of cluster and final check of validity of the FAT sub-type */
03216     n_clst = (n_vol - n_rsv - n_fat * N_FATS - n_dir) / au;
03217     if (   (fmt == FS_FAT16 && n_clst < MIN_FAT16)
03218         || (fmt == FS_FAT32 && n_clst < MIN_FAT32))
03219         return FR_MKFS_ABORTED;
03220 
03221     /* Create partition table if required */
03222     if (sfd == 1) {
03223         md = 0xF0;
03224     } else {

03226 
03227         mem_set(fs->win, 0, SS(fs));
03228         tbl = fs->win+MBR_Table;

03230         if (n_disk < 63UL * 255 * 1024) {   /* Partition end in CHS */
03231             n_disk = n_disk / 63 / 255;
03232             tbl[7] = (BYTE)n_disk;
03233             tbl[6] = (BYTE)((n_disk >> 2) | 63);
03234         } else {
03235             ST_WORD(&tbl[6], 0xFFFF);
03236         }
03237         tbl[5] = 254;
03238         if (fmt != FS_FAT32)            /* System ID */
03239             tbl[4] = (n_vol < 0x10000) ? 0x04 : 0x06;
03240         else
03241             tbl[4] = 0x0c;
03242         ST_DWORD(tbl+8, 63);            /* Partition start in LBA */
03243         ST_DWORD(tbl+12, n_vol);        /* Partition size in LBA */
03244         ST_WORD(tbl+64, 0xAA55);        /* Signature */
03245         if (disk_write(drv, fs->win, 0, 1) != RES_OK)
03246             return FR_DISK_ERR;
03247         md = 0xF8;
03248     }
03249 
03250     /* Create VBR */
03251     tbl = fs->win;                              /* Clear buffer */
03252     mem_set(tbl, 0, SS(fs));
03253     ST_DWORD(tbl+BS_jmpBoot, 0x90FEEB);         /* Boot code (jmp $, nop) */
03254     as = SS(fs);                                /* Sector size */
03255     ST_WORD(tbl+BPB_BytsPerSec, as);
03256     tbl[BPB_SecPerClus] = (BYTE)au;             /* Sectors per cluster */
03257     ST_WORD(tbl+BPB_RsvdSecCnt, n_rsv);         /* Reserved sectors */
03258     tbl[BPB_NumFATs] = N_FATS;                  /* Number of FATs */
03259     as = (fmt == FS_FAT32) ? 0 : N_ROOTDIR;     /* Number of rootdir entries */
03260     ST_WORD(tbl+BPB_RootEntCnt, as);
03261     if (n_vol < 0x10000) {                      /* Number of total sectors */
03262         ST_WORD(tbl+BPB_TotSec16, n_vol);
03263     } else {
03264         ST_DWORD(tbl+BPB_TotSec32, n_vol);
03265     }
03266     tbl[BPB_Media] = md;                        /* Media descriptor */
03267     ST_WORD(tbl+BPB_SecPerTrk, 63);             /* Number of sectors per track */
03268     ST_WORD(tbl+BPB_NumHeads, 255);             /* Number of heads */
03269     ST_DWORD(tbl+BPB_HiddSec, b_vol);           /* Hidden sectors */
03270     n = get_fattime();                          /* Use current time as VSN */
03271     if (fmt == FS_FAT32) {
03272         ST_DWORD(tbl+BS_VolID32, n);            /* VSN */
03273         ST_DWORD(tbl+BPB_FATSz32, n_fat);       /* Number of sectors per FAT */
03274         ST_DWORD(tbl+BPB_RootClus, 2);          /* Root directory start cluster (2) */
03275         ST_WORD(tbl+BPB_FSInfo, 1);             /* FSInfo record offset (VBR+1) */
03276         ST_WORD(tbl+BPB_BkBootSec, 6);          /* Backup boot record offset (VBR+6) */
03277         tbl[BS_DrvNum32] = 0x80;                /* Drive number */
03278         tbl[BS_BootSig32] = 0x29;               /* Extended boot signature */
03279         mem_cpy(tbl+BS_VolLab32, "NO NAME    FAT32   ", 19);    /* Volume label, FAT signature */
03280     } else {
03281         ST_DWORD(tbl+BS_VolID, n);              /* VSN */
03282         ST_WORD(tbl+BPB_FATSz16, n_fat);        /* Number of sectors per FAT */
03283         tbl[BS_DrvNum] = 0x80;                  /* Drive number */
03284         tbl[BS_BootSig] = 0x29;                 /* Extended boot signature */
03285         mem_cpy(tbl+BS_VolLab, "NO NAME    FAT     ", 19);  /* Volume label, FAT signature */
03286     }
03287     ST_WORD(tbl+BS_55AA, 0xAA55);               /* Signature (Offset is fixed here regardless of sector size) */
03288     if (disk_write(drv, tbl, b_vol, 1) != RES_OK)   /* Original (VBR) */
03289         return FR_DISK_ERR;
03290     if (fmt == FS_FAT32)                        /* Backup (VBR+6) */
03291         disk_write(drv, tbl, b_vol + 6, 1);
03292 
03293     /* Initialize FAT area */
03294     for (i = 0; i < N_FATS; i++) {
03295         mem_set(tbl, 0, SS(fs));            /* 1st sector of the FAT  */

03297         if (fmt != FS_FAT32) {
03298             n |= (fmt == FS_FAT12) ? 0x00FFFF00 : 0xFFFFFF00;
03299             ST_DWORD(tbl+0, n);             /* Reserve cluster #0-1 (FAT12/16) */
03300         } else {
03301             n |= 0x0FFFFF00;

03303             ST_DWORD(tbl+4, 0x0FFFFFFF);
03304             ST_DWORD(tbl+8, 0x0FFFFFFF);    /* Reserve cluster #2 for root dir */
03305         }
03306         if (disk_write(drv, tbl, b_fat++, 1) != RES_OK)
03307             return FR_DISK_ERR;

03310             if (disk_write(drv, tbl, b_fat++, 1) != RES_OK)
03311                 return FR_DISK_ERR;
03312         }
03313     }
03314 
03315     /* Initialize root directory */

03330         disk_write(drv, tbl, b_vol + 7, 1); /* Backup  (VBR+7) */
03331     }
