00043 /**
00044  * \addtogroup SAMGUI_WGT_CORE_FRONTEND WGT Core Frontend
00045  * @{
00046  *   \addtogroup SAMGUI_WGT_Touchscreen_frontend SAM-GUI WGT ADS7843 Frontend
00047  *   @{
00048  * The Touchscreen driver posts messages to the Widget message queue:
00049  * <ul>
00050  * <li>WGT_MSG_POINTER_RAW: pen pressed, provides uncalibrated-filtered pen position. </li>
00051  * <li>WGT_MSG_POINTER_PRESSED: pen pressed, provides calibrated-filtered pen position.</li>
00052  * <li>WGT_MSG_POINTER: pen moved, provides calibrated-filtered pen position.</li>
00053  * <li>WGT_MSG_POINTER_RELEASED: pen released, provides calibrated-filtered pen position.</li>
00054  * \n
00055  *
00056  * Touchscreen ADS7843 is connected to the microcontroller through SPI and an
00057  * a signal connected to a PIO corresponding to the pen press detection.
00058  * Each time a pen press is detected, the microcontroller is notified through
00059  * the interrupt line triggering a PIO interrupt.
00060  * \n
00061  *
00062  * The touchscreen driver starts a task suspended until the PIO interrupt
00063  * handler releases a semaphore. Then the task does measurements until the pen
00064  * is detected as released.
00065  * Once the pen is released, the task enters in suspend mode, waiting to get the
00066  * semaphore ie. waiting for the next pen press detection.
00067  *
00068  * Related files :\n
00069  * \ref ads7843.c\n
00070  * \ref ads7843.h\n
00071 */

00095 /**
00096   * \brief Returns Touchscreen pen position
00097   * Performs several (NB_SAMPLES succesive measurements and do a mean. Then
00098   * calibration function is applied according to the calibration parameters.
00099 */
00100 static void _WFE_ADS7843_GetPenPosition( uint32_t *pdwX, uint32_t *pdwY )
00101 {
00102     uint32_t i ;
00103     uint32_t dwX ; /* Measured coordinates */
00104     uint32_t dwY ; /* Measured coordinates */
00105     uint32_t dwTrueX ; /* Filtered coordinates */
00106     uint32_t dwTrueY ; /* Filtered coordinates */
00107 
00108 
00109     /* Do several successive measurements and process a mean */
00110     dwTrueX = 0 ;
00111     dwTrueY = 0 ;
00112     for ( i = 0; i < NB_SAMPLES; ++i )
00113     {
00114         ADS7843_GetPosition( &dwX, &dwY ) ;
00115         dwTrueX += dwX;
00116         dwTrueY += dwY;
00117     }
00118     dwTrueX /= NB_SAMPLES;
00119     dwTrueY /= NB_SAMPLES;
00120 
00121     /* If calibration has been done, apply calibration function */
00122     if ( gTsCalibrated )
00123     {
00124         dwTrueX = gs_ADS7843CalibParameters.dwPointX
00125                 - ((int32_t)(gs_ADS7843CalibParameters.dwMeasureX - (int32_t)dwTrueX) * 1024)
00126                 / gs_ADS7843CalibParameters.lSlopeX ;
00127         dwTrueY =  gs_ADS7843CalibParameters.dwPointY
00128                 - ((int32_t)(gs_ADS7843CalibParameters.dwMeasureY - (int32_t)dwTrueY) * 1024)
00129                 / gs_ADS7843CalibParameters.lSlopeY ;
00130 
00131         if ((int32_t)dwTrueX < 0 ) /* Is pPoint[0] negative ? */
00132             dwTrueX = 0;
00133         if ( dwTrueX > BOARD_LCD_WIDTH ) /* Is pPoint[0] bigger than the LCD width ? */
00134             dwTrueX = BOARD_LCD_WIDTH ;
00135 
00136         if ((int32_t)dwTrueY < 0 ) /* Is pPoint[0] negative ? */
00137             dwTrueY = 0;
00138         if ( dwTrueY > BOARD_LCD_HEIGHT ) /* Is pPoint[0] bigger than the LCD width ? */
00139             dwTrueY = BOARD_LCD_HEIGHT ;
00140     }
00141 
00142     /* Update pen position coordinates */
00143     *pdwX = dwTrueX;
00144     *pdwY = dwTrueY;
00145 }

00147 /**
00148   * \brief Touch screen sampling task
00149   *
00150   * This task is resumed when a PIO interrupt release the semaphore, that's to
00151   * say when the pen is pressed. Then this task reports pen activity each
00152   * SAMPLING_RATE period until the pen is released
00153 */
00154 static void _WFE_ADS7843_Task( void* pParameter )
00155 {
00156     uint8_t  isPenPressed;
00157     uint32_t dwX, dwPrevX;
00158     uint32_t dwY, dwPrevY;
00159 
00160     while ( 1 )
00161     {
00162         /* Block waiting for the semaphore to become available: */
00163         /* A PIO interrupt has been triggered: a key has been pressed */
00164         if ( xSemaphoreTake( gTsSemaphore, portMAX_DELAY ) != pdTRUE )
00165         {
00166             continue ;
00167         }
00168 
00169         /* Add a delay between interrupt and data acquisition */
00170         vTaskDelay( 5 ) ;
00171 
00172         isPenPressed = 1 ;
00173 
00174         /* Perform a first measurement corresponding to the key pressed */
00175         _WFE_ADS7843_GetPenPosition( &dwX, &dwY ) ;
00176 
00177         /* If it is not calibrated send the raw value to the application */
00178         if ( !gTsCalibrated )
00179         {
00180             WGT_SendMessageISR( WGT_MSG_POINTER_RAW, dwX, dwY ) ;
00181             isPenPressed = 0 ;
00182         }
00183         else
00184         {
00185             /* Send the key pressed position */
00186             WGT_SendMessageISR( WGT_MSG_POINTER_PRESSED, dwX, dwY) ;
00187             dwPrevX = dwX ;
00188             dwPrevY = dwY ;
00189         }
00190 
00191         /* Perform measurements at SAMPLING_RATE while the key is pressed */
00192         while ( isPenPressed )
00193         {
00194             vTaskDelay( SAMPLING_PERIOD/portTICK_RATE_MS ) ;
00195 
00196             /* Perform measurements at SAMPLING_RATE */
00197             _WFE_ADS7843_GetPenPosition( &dwX, &dwY ) ;
00198 
00199             /* If the touch has been released exit the loop and post release message */
00200             if ( PIO_Get( &_WFE_ADS7843_pinPenIRQ ) == 1 )
00201             {
00202                 WGT_SendMessageISR( WGT_MSG_POINTER_RELEASED, dwPrevX, dwPrevY ) ;
00203                 isPenPressed = 0 ;
00204             }
00205             /* Else send the pen position if this one is different from the previous one */
00206             else
00207             {
00208                 if ( (dwX != dwPrevX) || (dwY != dwPrevY) )
00209                 {
00210                     WGT_SendMessageISR( WGT_MSG_POINTER, dwX, dwY ) ;
00211                     dwPrevX = dwX ;
00212                     dwPrevY = dwY ;
00213                 }
00214             }
00215 
00216         }
00217 
00218         /* Loop and wait for PIO ISR to release the semaphore */
00219         /* Enable the PIO interrupt */
00220         PIO_EnableIt( &_WFE_ADS7843_pinPenIRQ ) ;
00221     }
00222 }
