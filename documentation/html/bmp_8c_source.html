00166 /// Loads a BMP image located at the given address, decodes it and stores the
00167 /// resulting image inside the provided buffer. Image must have the specified
00168 /// width & height.
00169 /// If no buffer is provided, this function simply checks if it is able to
00170 /// decode the image.
00171 /// \param file  Buffer which holds the BMP file.
00172 /// \param buffer  Buffer in which to store the decoded image.
00173 /// \param width  Buffer width in pixels.
00174 /// \param height  Buffer height in pixels.
00175 /// \param bpp  Number of bits per pixels that the buffer stores.
00176 /// \return 0 if the image has been loaded; otherwise returns an error code.
00177 //------------------------------------------------------------------------------
00178 uint8_t BMP_Decode( void *file, uint8_t *buffer, uint32_t width, uint32_t height, uint8_t bpp )
00179 {
00180     BMPHeader *header;
00181     uint32_t i, j;
00182     uint8_t r, g, b;
00183     uint8_t *image;
00184 
00185     // Read header information
00186     header = (BMPHeader*) file;
00187 
00188     // Verify that the file is valid
00189     if ( !BMP_IsValid( file ) )
00190     {
00191         TRACE_ERROR("BMP_Decode: File type is not 'BM' (0x%04X).\n\r",header->type);
00192 
00193         return 1;
00194     }
00195 
00196     // Check that parameters match
00197     if ( (header->compression != 0) || (header->width != width) || (header->height != height))
00198     {
00199         TRACE_ERROR("BMP_Decode: File format not supported\n\r");
00200         TRACE_ERROR(" -> .compression = %u\n\r", (unsigned int)header->compression);
00201         TRACE_ERROR(" -> .width = %u\n\r", (unsigned int)header->width);
00202         TRACE_ERROR(" -> .height = %u\n\r", (unsigned int)header->height);
00203         TRACE_ERROR(" -> .bits = %d\n\r", (int)header->bits);
00204 
00205         return 2;
00206     }
00207 
00208     // Get image data
00209     image = (uint8_t *) ((uint32_t) file + header->offset);
00210 
00211     // Check that the bpp resolution is supported
00212     // Only a 24-bit output & 24- or 8-bit input are supported
00213     if ( bpp != 24 )
00214     {
00215         TRACE_ERROR("BMP_Decode: Output resolution not supported\n\r");
00216 
00217         return 3;
00218     }
00219     else
00220     {
00221         if (header->bits == 24)
00222         {
00223             // Decoding is ok
00224             if (!buffer) return 0;
00225 
00226             // Get image data (swapping red & blue)
00227             for ( i=0 ; i < height ; i++ )
00228             {
00229                 for ( j=0 ; j < width; j++ )
00230                 {
00231                     r = image[((height - i - 1) * width + j) * 3 + 2];
00232                     g = image[((height - i - 1) * width + j) * 3 + 1];
00233                     b = image[((height - i - 1) * width + j) * 3];
00234 
00235     #if defined(BOARD_LCD_RGB565)
00236                     // Interlacing
00237                     r = ((r << 1) & 0xF0) | ((g & 0x80) >> 4) | ((r & 0x80) >> 5);
00238                     g = (g << 1) & 0xF8;
00239                     b = b & 0xF8;
00240 
00241                     buffer[(i * width + j) * 3] = b;
00242                     buffer[(i * width + j) * 3 + 1] = g;
00243                     buffer[(i * width + j) * 3 + 2] = r;
00244 
00245     #else
00246                     buffer[(i * width + j) * 3] = r;
00247                     buffer[(i * width + j) * 3 + 1] = g;
00248                     buffer[(i * width + j) * 3 + 2] = b;
00249     #endif //#if defined(BOARD_LCD_RGB565)
00250                 }
00251             }
00252         }
00253         else
00254         {
00255             if ( header->bits == 8 )
00256             {
00257                 BMPPaletteEntry palette[256];
00258 
00259                 // Decoding is ok
00260                 if (!buffer) return 0;
00261 
00262                 // Retrieve palette
00263                 memcpy( palette, (uint8_t *) ((uint32_t) file + sizeof( BMPHeader )), header->offset - sizeof( BMPHeader ) ) ;
00264 
00265                 // Decode image (reversing row order)
00266                 for ( i=0 ; i < height ; i++ )
00267                 {
00268                     for (j=0; j < width; j++)
00269                     {
00270                         r = palette[image[(height - i - 1) * width + j]].r;
00271                         g = palette[image[(height - i - 1) * width + j]].g;
00272                         b = palette[image[(height - i - 1) * width + j]].b;
00273 
00274                         buffer[(i * width + j) * 3] = r;
00275                         buffer[(i * width + j) * 3 + 1] = g;
00276                         buffer[(i * width + j) * 3 + 2] = b;
00277                     }
00278                 }
00279             }
00280             else
00281             {
00282 
00283                 TRACE_ERROR("BMP_Decode: Input resolution not supported\n\r");
00284                 TRACE_INFO("header->bits 0x%X \n\r", header->bits);
00285                 return 4 ;
00286             }
00287         }
00288     }
00289 
00290     return 0 ;
00291 }

00294 /// Convert RGB 565 to RGB 555 (RGB 555 is adapted to LCD)
00295 /// \param fileSource  Buffer which holds the RGB file
00296 /// \param fileDestination  Buffer in which to store the decoded image
00297 /// \param width  Buffer width in pixels.
00298 /// \param height  Buffer height in pixels.
00299 /// \param bpp  Number of bits per pixels that the buffer stores.
00300 //------------------------------------------------------------------------------
00301 void RGB565toBGR555( uint8_t *fileSource, uint8_t *fileDestination, uint32_t width, uint32_t height, uint8_t bpp )
00302 {
00303     uint32_t i;
00304     uint32_t j;
00305     uint32_t row;
00306 
00307     for (i=0; i < height*(bpp/8); i++)
00308     {
00309         row = (i*width*(bpp/8));
00310 
00311         for (j=0; j <= width*(bpp/8); j+=2)
00312         {
00313             fileDestination[row+j] = ((fileSource[row+j+1]>>3)&0x1F)
00314                                     | (fileSource[row+j]&0xE0);
00315             fileDestination[row+j+1] = (fileSource[row+j+1]&0x03)
00316                                     | ((fileSource[row+j]&0x1F)<<2);
00317         }
00318     }
