00127 /**
00128  * \brief This is the code that gets called on processor reset.
00129  * To initialize the device, and call the main() routine.
00130  */
00131 void ResetException( void )
00132 {
00133     uint32_t *pSrc, *pDest ;
00134 
00135     /* Low level Initialize */
00136     LowLevelInit() ;
00137 
00138     /* Initialize the relocate segment */
00139     pSrc = &_etext ;
00140     pDest = &_srelocate ;
00141 
00142     if ( pSrc != pDest )
00143     {
00144         for ( ; pDest < &_erelocate ; )
00145         {
00146             *pDest++ = *pSrc++ ;
00147         }
00148     }
00149 
00150     /* Clear the zero segment */
00151     for ( pDest = &_szero ; pDest < &_ezero ; )
00152     {
00153         *pDest++ = 0;
00154     }
00155 
00156     /* Set the vector table base address */
00157     pSrc = (uint32_t *)&_sfixed;
00158     SCB->VTOR = ( (uint32_t)pSrc & SCB_VTOR_TBLOFF_Msk ) ;
00159     
00160     if ( ((uint32_t)pSrc >= IRAM_ADDR) && ((uint32_t)pSrc < IRAM_ADDR+IRAM_SIZE) )
00161     {
00162         SCB->VTOR |= 1 << SCB_VTOR_TBLBASE_Pos ;
00163     }
00164 
00165     /* Initialize the C library */
00166     __libc_init_array() ;
00167 
00168     /* Branch to main function */
00169     main() ;
00170 
00171     /* Infinite loop */
00172     while ( 1 ) ;
00173 }
