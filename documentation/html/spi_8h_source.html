00062 /** Calculate the PCS field value given the chip select NPCS value */
00063 #define SPI_PCS(npcs)       ((~(1 << npcs) & 0xF) << 16)
00064 
00065 /** Calculates the value of the CSR SCBR field given the baudrate and MCK. */
00066 #define SPI_SCBR(baudrate, masterClock) ((uint32_t) (masterClock / baudrate) << 8)
00067 
00068 /** Calculates the value of the CSR DLYBS field given the desired delay (in ns) */
00069 #define SPI_DLYBS(delay, masterClock) ((uint32_t) (((masterClock / 1000000) * delay) / 1000) << 16)
00070 
00071 /** Calculates the value of the CSR DLYBCT field given the desired delay (in ns) */
00072 #define SPI_DLYBCT(delay, masterClock) ((uint32_t) (((masterClock / 1000000) * delay) / 32000) << 24)
