00030 /**
00031  * \page pmc_clock_switching PMC Clock Switching Example.
00032  *
00033  * \section Purpose
00034  * This example shows how to switch from a clock to another (PLLA, PLLB, SLCK, MAINCK)
00035  * or change divider.
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with SAM3S evaluation kits.
00040  *
00041  * \section Description
00042  *
00043  * Upon startup, the program configure pios for UART,PCK and botton. The baud rate of
00044  * UART is configured as 2400 bps. The application does some printf with the current
00045  * configuration (except 32KHz slow clock ) and waits for button pressed to switch to
00046  * next configuration. PCK1 Outputs can be selected from the clocks provided by the
00047  * clock (PLLA, PLLB, SLCK, MAINCK) and driven on the pin PCK1 on PA17 (Peripheral B).
00048  * After the clock switchs, the PCK1 output signal can be measured by scope compare with
00049  * the clock configuration.
00050  *
00051  * <ul>
00052  * <li> The Clock Generator integrates a 32,768 Hz low-power oscillator.
00053  * In order to use this oscillator, the XIN32/PA7 and XOUT32/P8 pins must be connected
00054  * to a 32,768 Hz crystal. The user can select the crystal oscillator to be the source of
00055  * the slow clock, as it provides a more accurate frequency. The command is made by
00056  * function _PmcEnableExternal32K().</li>
00057  * <li> MAINCK is the output of the Main Clock Oscillator selection: either the Crystal
00058  * Oscillator or 4/8/12 MHz Fast RC Oscillator. The user can select the output frequency
00059  * of the Fast RC Oscillator: either 4 MHz, 8 MHz or 12 MHz are available. It can be done
00060  * through function _PmcMainClockSwitchFastRc(). The user can also select the 3 to 20 MHz Crystal
00061  * oscillator to be the source of MAINCK, as it provides a more accurate frequency.
00062  * The function _PmcMainClockSwitchMainOsc() show how to enable the oscillator. </li>
00063  * <li> The PMC features one Divider/PLL Blocks that permit a wide range of frequencies to be
00064  * selected on either the master clock, the processor clock or the programmable clock outputs.
00065  * Function _PmcConfigurePllClock() descripts the deatil how to configure PLLA/B.</li>
00066  * </ul>
00067  * \section Usage
00068  *
00069  * -# Build the program and download it inside the evaluation board. Please
00070  *    refer to the
00071  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6224.pdf">
00072  *    SAM-BA User Guide</a>, the
00073  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00074  *    GNU-Based Software Development</a> application note or to the
00075  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00076  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00077  * -# On the computer, open and configure a terminal application
00078  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00079  *   - 2400 bauds
00080  *   - 8 bits of data
00081  *   - No parity
00082  *   - 1 stop bit
00083  *   - No flow control
00084  * -# Start the application.
00085  * -# In the terminal window, the following text should appear:
00086  *    \code
00087  *     -- Pmc clock Switching Example xxx --
00088  *     -- xxxxxx-xx
00089  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00090  *    \endcode
00091  * -# Press one of the keys listed in the menu to perform the corresponding action.
00092  *
00093  * \section References
00094  * - pmc_clock_switching/main.c
00095  */

00209 static void _PmcEnableFastRC( uint32_t dwRC )
00210 {
00211     uint32_t dwTimeout ;
00212 
00213     /* Enable Fast RC oscillator but DO NOT switch to RC now . Keep MOSCSEL to 1 */
00214     PMC->CKGR_MOR = CKGR_MOR_KEY( 0x37 ) | CKGR_MOR_MOSCSEL | CKGR_MOR_MOSCXTEN | CKGR_MOR_MOSCRCEN | dwRC ;

00217     for ( dwTimeout=0 ; !(PMC->PMC_SR & PMC_SR_MOSCRCS) && (dwTimeout++ < CLOCK_TIMEOUT) ; ) ;

00259     uint32_t timeout = 0;

00261     /* Select external slow clock */
00262     if ( (SUPC->SUPC_SR & SUPC_SR_OSCSEL) != SUPC_SR_OSCSEL_CRYST )
00263     {
00264         /* writing the Supply Controller Control register (SUPC_CR) with the XTALSEL bit at 1*/

00309 /**
00310  * \brief Selection of Master Clock.
00311  *
00312  * \remarks See chapter 26.13 in SAM3S datasheet.
00313  *
00314  * \param css  The Master Clock divider source.
00315  * \param pres Master Clock prescaler.
00316  */
00317 static void _PmcMasterClockSelection( uint32_t dwClockSource, uint32_t dwPrescaler )
00318 {
00319     uint32_t dwTimeout = 0 ;
00320 
00321     switch ( dwClockSource )
00322     {
00323         case PMC_MCKR_CSS_SLOW_CLK :
00324         case PMC_MCKR_CSS_MAIN_CLK :
00325             /* The Master Clock selection is made by writing the CSS field in PMC_MCKR programs the prescaler.*/
00326             PMC->PMC_MCKR = (PMC->PMC_MCKR & ~PMC_MCKR_CSS_Msk) | dwClockSource ;
00327 
00328             /* Once the PMC_MCKR register has been written, the user must wait for the MCKRDY bit to be set in the PMC_SR register.*/
00329             while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) && (dwTimeout++ < CLOCK_TIMEOUT) ) ;
00330 
00331             /* The Master Clock selection is made by writing the PRES field in PMC_MCKR programs the prescaler.*/
00332             PMC->PMC_MCKR = (PMC->PMC_MCKR & ~PMC_MCKR_PRES_Msk) | dwPrescaler ;
00333 
00334             /* Once the PMC_MCKR register has been written, the user must wait for the MCKRDY bit to be set in the PMC_SR register.*/
00335             while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) && (dwTimeout++ < CLOCK_TIMEOUT) ) ;
00336         break ;
00337 
00338         case PMC_MCKR_CSS_PLLA_CLK :
00339         case PMC_MCKR_CSS_PLLB_CLK :
00340             /* The Master Clock selection is made by writing the PRES field in PMC_MCKR programs the prescaler.*/
00341             PMC->PMC_MCKR = (PMC->PMC_MCKR & ~PMC_MCKR_PRES_Msk) | dwPrescaler ;
