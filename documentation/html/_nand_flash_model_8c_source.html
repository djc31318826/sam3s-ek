00044 /**
00045  * \brief  Get the power of input, given a certain result, i.e. input^(power) = result.
00046  *
00047  * \param result  a certain output we want to calculate.
00048  * \param input  the input of the power.
00049  * \return the value of "power" if succesfully find the power.
00050  */
00051 #if defined(OP_BOOTSTRAP_on)
00052 unsigned int CALPOW(unsigned int result, unsigned int input)
00053 {
00054     unsigned int i=0;
00055 
00056     while(i<32)
00057     {
00058         if(result == (input << i))
00059             return i;
00060         i++;
00061     }
00062 
00063     return 0;
00064 }
00065 #endif

00067 /**
00068  * \brief  Get the interger part of input, given a certain result, i.e.  return = result / input.
00069  *
00070  * \param result  a certain output we want to calculate.
00071  * \param input  the input of the division.
00072  * \return the value of interger part of the result/input.

00074 #if defined(OP_BOOTSTRAP_on)
00075 unsigned int CALINT(unsigned int result, unsigned int input)
00076 {
00077     unsigned int i=0;
00078     unsigned int tmpInput=0;
00079

00081     {
00082         tmpInput +=input;
00083         i++;
00084         if(tmpInput == result)
00085           return i;
00086         else if (tmpInput > result)
00087           return (i-1);
00088     }

00101  * \param modelList  List of NandFlashModel instances.

00188 /**
00189  * \brief Translates address/size access of a NandFlashModel to block, page and offset values.
00190  *
00191  * \param modelList  List of NandFlashModel instances.
00192  * \param address  Access address.
00193  * \param size  Access size in bytes.
00194  * \param block  Stores the first accessed block number.
00195  * \param page  Stores the first accessed page number inside the first block.
00196  * \param offset  Stores the byte offset inside the first accessed page.
00197  * \return 0 if the access is correct; otherwise returns NandCommon_ERROR_OUTOFBOUNDS.
00198  * \note The values are stored in the provided variables if their pointer is not 0.
00199  */
00200 unsigned char NandFlashModel_TranslateAccess(
00201     const struct NandFlashModel *model,
00202     unsigned int address,
00203     unsigned int size,
00204     unsigned short *block,
00205     unsigned short *page,
00206     unsigned short *offset)
00207 {
00208      /* Check that access is not too big */
00209     #if !defined(OP_BOOTSTRAP_on)
00210     if ((address + size) > NandFlashModel_GetDeviceSizeInBytes(model)) {
00211 
00212         TRACE_DEBUG("NandFlashModel_TranslateAccess: out-of-bounds access.\n\r");
00213         return NandCommon_ERROR_OUTOFBOUNDS;
00214     }
00215     #endif
00216 
00217     /* Get Nand info */
00218     unsigned int blockSize = NandFlashModel_GetBlockSizeInBytes(model);
00219     unsigned int pageSize = NandFlashModel_GetPageDataSize(model);
00220 
00221     /* Translate address*/
00222     #if !defined(OP_BOOTSTRAP_on)
00223     unsigned short tmpBlock = address / blockSize;
00224     address -= tmpBlock * blockSize;
00225     unsigned short tmpPage = address / pageSize;
00226     address -= tmpPage * pageSize;
00227     unsigned short tmpOffset = address;
00228     #else
00229     unsigned short tmpBlock = CALINT(address, blockSize);
00230     address -= tmpBlock * blockSize;
00231     unsigned short tmpPage = CALINT(address, pageSize);
00232     address -= tmpPage * pageSize;
00233     unsigned short tmpOffset= address;
00234     #endif
00235 
00236     /* Save results */
00237     if (block) {
00238 
00239         *block = tmpBlock;
00240     }
00241     if (page) {
00242 
00243         *page = tmpPage;
00244     }
00245     if (offset) {
00246 
00247         *offset = tmpOffset;
00248     }
00249 
00250     return 0;
00251 }
