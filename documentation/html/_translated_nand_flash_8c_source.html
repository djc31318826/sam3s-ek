00135 /**
00136  * \brief  Allocates the best-fitting physical block for the given logical block.
00137  *
00138  * \param translated  Pointer to a TranslatedNandFlash instance.
00139  * \param block  Logical block number.
00140  * \return 0 if successful; otherwise returns NandCommon_ERROR_NOBLOCKFOUND if
00141  * there are no more free blocks, or a NandCommon_ERROR code.
00142  */
00143 static unsigned char AllocateBlock(
00144     struct TranslatedNandFlash *translated,
00145     unsigned short block)
00146 {
00147     unsigned short freeBlock, liveBlock;
00148     unsigned char error;
00149     signed int eraseDifference;
00150 
00151     TRACE_DEBUG("Allocating a new block\n\r");
00152 
00153     /* Find youngest free block and youngest live block*/
00154     if (ManagedNandFlash_FindYoungestBlock(MANAGED(translated),
00155                                            NandBlockStatus_FREE,
00156                                            &freeBlock)) {
00157 
00158         TRACE_ERROR("AllocateBlock: Could not find a free block\n\r");
00159         return NandCommon_ERROR_NOBLOCKFOUND;
00160     }
00161 
00162     /* If this is the last free block, save the logical mapping in it and clean dirty blocks */
00163     TRACE_DEBUG("Number of FREE blocks: %d\n\r",
00164               ManagedNandFlash_CountBlocks(MANAGED(translated), NandBlockStatus_FREE));
00165     if (ManagedNandFlash_CountBlocks(MANAGED(translated),
00166                                      NandBlockStatus_FREE) == 1) {
00167 
00168         /* Save mapping and clean dirty blocks*/
00169         TRACE_DEBUG("Last FREE block, cleaning up ...\n\r");
00170 
00171         error = MappedNandFlash_SaveLogicalMapping(MAPPED(translated), freeBlock);
00172         if (error)
00173         {
00174             TRACE_ERROR("AllocateBlock: Failed to save mapping\n\r");
00175             return error;
00176         }
00177         error = ManagedNandFlash_EraseDirtyBlocks(MANAGED(translated));
00178         if (error)
00179         {
00180             TRACE_ERROR("AllocatedBlock: Failed to erase dirty blocks\n\r");
00181             return error;
00182         }
00183 
00184         /* Allocate new block*/
00185         return AllocateBlock(translated, block);
00186     }
00187 
00188     /* Find youngest LIVE block to check the erase count difference*/
00189     if (!ManagedNandFlash_FindYoungestBlock(MANAGED(translated),
00190                                             NandBlockStatus_LIVE,
00191                                             &liveBlock))
00192     {
00193         /* Calculate erase count difference*/
00194         TRACE_DEBUG("Free block erase count = %d\n\r", MANAGED(translated)->blockStatuses[freeBlock].eraseCount);
00195         TRACE_DEBUG("Live block erase count = %d\n\r", MANAGED(translated)->blockStatuses[liveBlock].eraseCount);
00196         eraseDifference = absv(MANAGED(translated)->blockStatuses[freeBlock].eraseCount
00197                               - MANAGED(translated)->blockStatuses[liveBlock].eraseCount);
00198 
00199         /* Check if it is too big*/
00200         if (eraseDifference > MAXERASEDIFFERENCE)
00201         {
00202             TRACE_WARNING("Erase difference too big, switching blocks\n\r");
00203             MappedNandFlash_Map(
00204                 MAPPED(translated),
00205                 MappedNandFlash_PhysicalToLogical(
00206                     MAPPED(translated),
00207                     liveBlock),
00208                 freeBlock);
00209             ManagedNandFlash_CopyBlock(MANAGED(translated),
00210                                        liveBlock,
00211                                        freeBlock);
00212 
00213             /* Allocate a new block*/
00214             return AllocateBlock(translated, block);
00215         }
00216     }
00217 
00218     /* Map block*/
00219     TRACE_DEBUG("Allocating PB#%d for LB#%d\n\r", freeBlock, block);
00220     MappedNandFlash_Map(MAPPED(translated), block, freeBlock);
00221 
00222     return 0;
00223 }

00270 /**
00271  * \brief  Reads the data and/or the spare area of a page on a translated nandflash.
00272  * If the block is not currently mapped but could be (i.e. there are available
00273  * physical blocks), then the data/spare is filled with 0xFF.
00274  *
00275  * \param translated  Pointer to a TranslatedNandFlash instance.
00276  * \param block  Logical block number.
00277  * \param page  Number of page to read inside logical block.
00278  * \param data  Data area buffer, can be 0.
00279  * \param spare  Spare area buffer, can be 0.
00280  * \return 0 if successful; otherwise returns NandCommon_ERROR_NOMOREBLOCKS
00281  */
00282 unsigned char TranslatedNandFlash_ReadPage( const struct TranslatedNandFlash *translated, unsigned short block,
00283                                             unsigned short page, void *data, void *spare )
00284 {
00285     unsigned char error ;
00286 
00287     TRACE_INFO("TranslatedNandFlash_ReadPage(B#%d:P#%d)\n\r", block, page);
00288 
00289     /* If the page to read is in the current block, there is a previous physical
00290        block and the page is clean -> read the page in the old block since the
00291        new one does not contain meaningful data*/
00292     if ( (block == translated->currentLogicalBlock) && (translated->previousPhysicalBlock != -1)
00293          && (PageIsClean(translated, page)) )
00294     {
00295         TRACE_DEBUG("Reading page from current block\n\r");
00296         return ManagedNandFlash_ReadPage( MANAGED( translated ), translated->previousPhysicalBlock,
00297                                           page, data, spare ) ;
00298     }
00299     else
00300     {
00301         /* Try to read the page from the logical block*/
00302         error = MappedNandFlash_ReadPage(MAPPED(translated), block, page, data, spare);
00303 
00304         /* Block was not mapped*/
00305         if ( error == NandCommon_ERROR_BLOCKNOTMAPPED )
00306         {
00307             assert( !spare ) ; /* "Cannot read the spare information of an unmapped block\n\r" */
00308 
00309             /* Check if a block can be allocated*/
00310             if ( BlockCanBeAllocated( translated ) )
00311             {
00312                 /* Return 0xFF in buffers with no error*/
00313                 TRACE_DEBUG("Block #%d is not mapped but can be allocated, filling buffer with 0xFF\n\r", block);
00314                 if (data)
00315                 {
00316                     memset(data, 0xFF, NandFlashModel_GetPageDataSize(MODEL(translated)));
00317                 }
00318                 if (spare)
00319                 {
00320                     memset(spare, 0xFF, NandFlashModel_GetPageSpareSize(MODEL(translated)));
00321                 }
00322             }
00323             else
00324             {
00325                 TRACE_ERROR("Block #%d is not mapped and there are no more blocks available\n\r", block);
00326                 return NandCommon_ERROR_NOMOREBLOCKS;
00327             }
00328         }
00329         /* Error*/
00330         else
00331         {
00332             if (error)
00333             {
00334                 return error;
00335             }
00336         }
00337     }
00338 
00339     return 0;
00340 }

00342 /**
00343  * \brief  Writes the data and/or spare area of a page on a translated nandflash.
00344  * Allocates block has needed to keep the wear even between all blocks.
00345  *
00346  * \param translated  Pointer to a TranslatedNandFlash instance.
00347  * \param block  Logical block number.
00348  * \param page  Number of page to read inside logical block.
00349  * \param data  Data area buffer, can be 0.
00350  * \param spare  Spare area buffer, can be 0.
00351  * \return 0 if successful; otherwise returns error code
00352  */
00353 unsigned char TranslatedNandFlash_WritePage( struct TranslatedNandFlash *translated, unsigned short block,
00354                                              unsigned short page, void *data, void *spare )
00355 {
00356     unsigned char allocate = 1;
00357     unsigned char error;
00358 
00359     TRACE_INFO("TranslatedNandFlash_WritePage(B#%d:P#%d)\n\r", block, page);
00360 
00361     /* A new block must be allocated unless:*/
00362     /* 1. the block is not mapped and there are no more blocks to allocate*/
00363     if (MappedNandFlash_LogicalToPhysical(MAPPED(translated), block) == -1)
00364     {
00365         /* Block is not mapped, check if it can be*/
00366         if (!BlockCanBeAllocated(translated))
00367         {
00368             TRACE_ERROR("TranslatedNandFlash_WritePage: Not enough free blocks\n\r");
00369             return NandCommon_ERROR_NOMOREBLOCKS;
00370         }
00371         TRACE_DEBUG("Allocate because block not mapped\n\r");
00372     }
00373     /* or 2. the block to write is the current one and the page to write is clean*/
00374     else
00375     {
00376         if (translated->currentLogicalBlock == block)
00377         {
00378             if (PageIsClean(translated, page))
00379             {
00380                 TRACE_DEBUG("NO allocate because write in current block\n\r");
00381                 allocate = 0;
00382             }
00383             else
00384             {
00385                 TRACE_DEBUG("Allocate because page DIRTY in current block\n\r");
00386             }
00387         }
00388         else
00389         {
00390             TRACE_DEBUG("Allocate because block is mapped and different from current block\n\r");
00391         }
00392     }
00393 
00394     /* Allocate block if needed*/
00395     if ( allocate )
00396     {
00397         /* Flush current block write (if any) and then allocate block*/
00398         error = TranslatedNandFlash_Flush(translated);
00399         if (error)
00400         {
00401             return error;
00402         }
00403         translated->previousPhysicalBlock = MappedNandFlash_LogicalToPhysical( MAPPED(translated), block ) ;
00404         TRACE_DEBUG("Previous physical block is now #%d\n\r", translated->previousPhysicalBlock ) ;
00405         error = AllocateBlock( translated, block ) ;
00406         if ( error )
00407         {
00408             return error;
00409         }
00410 
00411         /* Block becomes the current block with all pages clean*/
00412         translated->currentLogicalBlock = block;
00413         MarkAllPagesClean(translated);
00414     }
00415 
00416     /* Start writing page*/
00417     error = MappedNandFlash_WritePage(MAPPED(translated),
00418                                       block,
00419                                       page,
00420                                       data,
00421                                       spare);
00422     if ( error )
00423     {
00424         return error;
00425     }
00426 
00427     /* If write went through, mark page as written*/
00428     MarkPageDirty(translated, page);
00429 
00430     return 0;
00431 }

00433 /**
00434  * \brief  Terminates the current write operation by copying all the missing pages from
00435  * the previous physical block.
00436  *
00437  * \param translated  Pointer to a TranslatedNandFlash instance.
00438  * \return 0 if successful; otherwise returns error code
00439  */
00440 unsigned char TranslatedNandFlash_Flush(struct TranslatedNandFlash *translated)
00441 {
00442     unsigned int i;
00443     unsigned char error;
00444     unsigned int currentPhysicalBlock;
00445 
00446     /* Check if there is a current block and a previous block*/
00447     if ((translated->currentLogicalBlock == -1)
00448         || (translated->previousPhysicalBlock == -1))
00449     {
00450         return 0;
00451     }
00452 
00453     TRACE_INFO("TranslatedNandFlash_Flush(PB#%d -> LB#%d)\n\r",
00454               translated->previousPhysicalBlock, translated->currentLogicalBlock);
00455 
00456     /* Copy missing pages in the current block*/
00457     currentPhysicalBlock = MappedNandFlash_LogicalToPhysical(
00458                                 MAPPED(translated),
00459                                 translated->currentLogicalBlock);
00460 
00461     for (i=0; i < NandFlashModel_GetBlockSizeInPages(MODEL(translated)); i++)
00462     {
00463         if (PageIsClean(translated, i))
00464         {
00465             TRACE_DEBUG("Copying back page #%d of block #%d\n\r", i, translated->previousPhysicalBlock);
00466 
00467             /* Copy page*/
00468             error = ManagedNandFlash_CopyPage(MANAGED(translated),
00469                                               translated->previousPhysicalBlock,
00470                                               i,
00471                                               currentPhysicalBlock,
00472                                               i);
00473             if (error)
00474             {
00475                 TRACE_ERROR("FinishCurrentWrite: copy page #%d\n\r", i);
00476                 return error;
00477             }
00478         }
00479     }
00480 
00481     translated->currentLogicalBlock = -1;
00482     translated->previousPhysicalBlock = -1;
